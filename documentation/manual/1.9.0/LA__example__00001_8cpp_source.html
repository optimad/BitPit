64int run(int rank, int nProcs)
65{
66 int nRows;
67 int nCols;
68 int nNZ;
69 if (nProcs > 1 && rank == 0) {
70 nRows = 0;
71 nCols = 0;
72 nNZ = 0;
73 } else {
74 nRows = 10;
75 nCols = 10;
76 nNZ = 10;
77 }
78
79 int blockSize = 1;
80
81 // Build matrix
82 log::cout() << "Building matrix..." << std::endl;
83
84 std::vector<long> rowPattern(1);
85 std::vector<double> rowValues(1);
86
87#if BITPIT_ENABLE_MPI==1
88 SparseMatrix matrix(MPI_COMM_WORLD, true, blockSize, nRows, nCols, nNZ);
89
90 int rowOffset = matrix.getRowGlobalOffset();
91 int colOffset = matrix.getColGlobalOffset();
92#else
93 SparseMatrix matrix(blockSize, nRows, nCols, nNZ);
94
95 int rowOffset = 0;
96 int colOffset = 0;
97
98#endif
99
100 for (int i = 0; i < nRows; ++i) {
101 rowPattern[0] = colOffset + i;
102 rowValues[0] = 1. / (double) (rowOffset + i + 1);
103
104 matrix.addRow(rowPattern, rowValues);
105 }
106
107 matrix.assembly();
108
109 // Build solver
110 log::cout() << "Building solver..." << std::endl;
111
112 SystemSolver solver;
113 solver.assembly(matrix);
114
115 double *rhs = solver.getRHSRawPtr();
116 for (int i = 0; i < nCols; ++i) {
117 rhs[i] = 1.;
118 }
119 solver.restoreRHSRawPtr(rhs);
120
121 double *initialSolution = solver.getSolutionRawPtr();
122 for (int i = 0; i < nRows; ++i) {
123 initialSolution[i] = 0;
124 }
125 solver.restoreSolutionRawPtr(initialSolution);
126
127 // Dump system
128 log::cout() << "Dumping initialized linear system..." << std::endl;
129
130 solver.dumpSystem("./data/", "example_initialized_linear_system_");
131
132 // Set KSP
133 KSPOptions &options = solver.getKSPOptions();
134 options.atol = 1e-50;
135 options.rtol = 1e-7;
136 options.maxits = 100;
137 options.subrtol = 1e-7;
138
139 // Solve System
140 log::cout() << "Solve linear system..." << std::endl;
141 solver.solve();
142
143 // Export solution
144 log::cout() << "Export solution..." << std::endl;
145 std::string solutionPath = std::string("./data/") + std::string("example_linear_system_solution.txt");
146 solver.exportSolution(solutionPath, bitpit::SystemSolver::FILE_BINARY);
147
148 return 0;
149}
