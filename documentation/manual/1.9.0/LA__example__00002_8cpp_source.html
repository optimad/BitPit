109 void preKSPSetupActions() override
110 {
111 //
112 // Preconditioner configuration
113 //
114 PCType preconditionerType;
115 if (m_multigrid) {
116 preconditionerType = PCGAMG;
117 } else {
118#if BITPIT_ENABLE_MPI == 1
119 if (this->isPartitioned()) {
120 preconditionerType = PCASM;
121 } else {
122 preconditionerType = PCILU;
123 }
124#else
125 preconditionerType = PCILU;
126#endif
127 }
128
129 PC preconditioner;
130 KSPGetPC(this->m_KSP, &preconditioner);
131 PCSetType(preconditioner, preconditionerType);
132
133 // Set preconditioner options
134 if (strcmp(preconditionerType, PCASM) == 0) {
135 if (this->m_KSPOptions.overlap != PETSC_DEFAULT) {
136 PCASMSetOverlap(preconditioner, this->m_KSPOptions.overlap);
137 }
138 } else if (strcmp(preconditionerType, PCILU) == 0) {
139 if (this->m_KSPOptions.levels != PETSC_DEFAULT) {
140 PCFactorSetLevels(preconditioner, this->m_KSPOptions.levels);
141 }
142 } else if (strcmp(preconditionerType, PCGAMG) == 0) {
143 log::cout() << "MULTIGRID SOLVER" << std::endl;
144 PetscReal thresholds[GAMG_MAX_LEVELS];
145 for (int i = 0; i < GAMG_MAX_LEVELS; ++i) {
146 thresholds[i] = GAMG_THRESHOLD;
147 }
148
149 PCGAMGSetNSmooths(preconditioner, 0);
150#if (PETSC_VERSION_MAJOR >= 3 && PETSC_VERSION_MINOR <= 17)
151#if BITPIT_ENABLE_MPI == 1
152 if (this->isPartitioned()) {
153 PCGAMGSetSymGraph(preconditioner, PETSC_TRUE);
154 }
155#endif
156#endif
157
158 PCGAMGSetNlevels(preconditioner, GAMG_MAX_LEVELS);
159#if (PETSC_VERSION_MAJOR >= 3 && PETSC_VERSION_MINOR >= 16)
160 PCGAMGSetThreshold(preconditioner, thresholds, GAMG_MAX_LEVELS);
161 PCGAMGSetThresholdScale(preconditioner, GAMG_THRESHOLD_SCALE);
162#else
163 PCGAMGSetThreshold(preconditioner, GAMG_THRESHOLD);
164#endif
165 }
166
167 //
168 // Solver configuration
169 //
170 KSPSetType(this->m_KSP, KSPFGMRES);
171 if (this->m_KSPOptions.restart != PETSC_DEFAULT) {
172 KSPGMRESSetRestart(this->m_KSP, this->m_KSPOptions.restart);
173 }
174 if (this->m_KSPOptions.rtol != PETSC_DEFAULT || this->m_KSPOptions.atol != PETSC_DEFAULT || this->m_KSPOptions.maxits != PETSC_DEFAULT) {
175 KSPSetTolerances(this->m_KSP, this->m_KSPOptions.rtol, this->m_KSPOptions.atol, PETSC_DEFAULT, this->m_KSPOptions.maxits);
176 }
177 KSPSetInitialGuessNonzero(this->m_KSP, PETSC_TRUE);
178 }

217int run(int rank, int nProcs)
218{
219 BITPIT_UNUSED(nProcs);
220
221 // Restore solver with initial guess
222 log::cout() << "Restoring linear system ..." << std::endl;
223 std::string solverPath = "./data/";
224 bool multigrid = false;
225 bool debug = false;
226 bool transpose = false;
227 MyLinearSolver restoredSolver(transpose, multigrid, debug);
228#if BITPIT_ENABLE_MPI==1
229 restoredSolver.restoreSystem(MPI_COMM_WORLD, solverPath, "example_initialized_linear_system_");
230#else
231 restoredSolver.restoreSystem(solverPath, "example_initialized_linear_system_");
232#endif
233 log::cout() << "Linear system restored." << std::endl;
234
235 // Check if multigrid is possible. Multigrid is not possible on BAIJ matrices
236 if (multigrid) {
237 log::cout() << "Enable multigrid preconditioner..." << std::endl;
238 int blockSize = restoredSolver.getBlockSize();
239 bool enableMultigrid = (blockSize == 1);
240 restoredSolver.setMultigridFlag(enableMultigrid);
241 if (enableMultigrid) {
242 log::cout() << "Multigrid preconditioner enabled..." << std::endl;
243 } else {
244 log::cout() << "The matrix is a block matrix, no multigrid preconditioning is possible." << std::endl;
245 }
246 }
247
248 // Set KSP options
249 KSPOptions &options = restoredSolver.getKSPOptions();
250 options.atol = 1e-50;
251 options.rtol = 1e-7;
252 options.maxits = 100;
253 options.subrtol = 1e-7;
254
255 // Solve linear system
256 log::cout() << "Solving linear system..." << std::endl;
257 restoredSolver.solve();
258 const KSPStatus &status = restoredSolver.getKSPStatus();
259 log::cout() << "Reason: " << status.convergence << " in its: " << status.its << std::endl;
260 log::cout() << "Linear system solved." << std::endl;
261
262 // Turn solution comparison on
263 bool compareSolutions = true;
264 if (compareSolutions) {
265 // Store expected solution for comparison
266 log::cout() << "Storing computed solution for comparison..." << std::endl;
267 std::size_t vecLength = restoredSolver.getRowElementCount();
268 std::vector<double> expectedSolution(vecLength, 0.0);
269 double *computedSolution = restoredSolver.getSolutionRawPtr();
270 for (std::size_t i = 0; i < vecLength; ++i) {
271 expectedSolution[i] = computedSolution[i];
272 }
273 restoredSolver.restoreSolutionRawPtr(computedSolution);
274
275 // Fill solution with solution from outside
276 log::cout() << "Restoring solution..." << std::endl;
277 std::string solutionPath = solverPath + "example_linear_system_solution.txt";
278 restoredSolver.fillSolution(solutionPath);
279 log::cout() << "Solution restored." << std::endl;
280
281 // Set comparison tolerance to the linear system relative tolerance
282 log::cout() << "Comparing solutions..." << std::endl;
283
284 const double TOLERANCE = options.rtol;
285 log::cout() << "Tolerance = " << options.rtol << std::endl;
286
287 // Compare computed solution with restored one
288 double *restoredSolution = restoredSolver.getSolutionRawPtr();
289 for (std::size_t i = 0; i < vecLength; ++i) {
290 // Check for absolute tolerance only
291 if (!utils::DoubleFloatingEqual()(expectedSolution[i] - restoredSolution[i], 0.0, TOLERANCE, TOLERANCE)) {
292 double error = expectedSolution[i] - restoredSolution[i];
293
294 log::cout() << " Solutions do not match (on rank #" << rank << " error for DoF # " << i << " is " << error << ")" << std::endl;
295 }
296 }
297 restoredSolver.restoreSolutionRawPtr(restoredSolution);
298 log::cout() << "Solutions compared..." << std::endl;
299 }
300
301 return 0;
302}
