753template <class T, size_t d>
754double norm(
755 const std::array<T, d> &x,
756 int p
757) {
758
759using namespace std;
760
761// ================================================================================== //
762// VARIABLES DECLARATION //
763// ================================================================================== //
764
765// Local variables
766double z = 0.0;
767double t, y;
768
769// Counters
770size_t i;
771int j;
772
773// ================================================================================== //
774// COMPUTE THE P-NORM //
775// ================================================================================== //
776if (p == 1) { return(norm1(x)); }
777if (p == 2) { return(norm2(x)); }
778
779if (d > 0) {
780 for (i = 0; i < d; i++) {
781 y = 1.0;
782 t = abs(x[i]);
783 for (j = 1; j <= p; j++) {
784 y = y*t;
785 } //next j
786 //z += pow(abs(x[i]), p);
787 z += y;
788 } //next i
789}
790
791return(std::exp(std::log(std::max(z, 1.0e-307))/((double) p))); };
