32// Octant header should be included before bitpit_communications header. If
33// this is not the case, octants will be streamed in the communication buffer
34// using the standard bitpit operator instead of the custom operator defined
35// for the octant. GCC will work also if the Octant header is included after
36// bitpit_communications header, but that's a bug in GCC.
37//
38// https://gcc.gnu.org/bugzilla/show_bug.cgi?id=51577
39// https://bugs.llvm.org/show_bug.cgi?id=47967
40//
41// For a dependent name used in a template definition, the lookup is postponed
42// until the template arguments are known, at which time ADL examines function
43// declarations with external linkage (until C++11) that are visible from the
44// template definition context as well as in the template instantiation context
45// (in other words, ADL will only look in namespaces associated with the
46// argument types), while unqualified lookup only examines function declarations
47// with external linkage (until C++11) that are visible from the template
48// definition context (in other words, adding a new function declaration after
49// template definition does not make it visible except via ADL). The stream
50// operator template is defined in the global namespace, whereas the octant
51// class is defined in the bitpit namespace. Hence, custom stream operator
52// defined for the octant class cannot be selected by ADL (ADL will only look
53// in the namaspace of its arguments, i.e., bitpit, whereas the custom stream
54// operator is defined in the global namespace). The only chance to select
55// the custom operator is through unqualified lookup, and for that to happen,
56// the stream operator defined for the octant class should be defined before
57// the template.

596 template<class Impl>
597 void
598 communicate(DataCommInterface<Impl> & userData){
599 //BUILD SEND BUFFERS
600 DataCommunicator communicator(m_comm);
601 size_t fixedDataSize = userData.fixedSize();
602 std::map<int,u32vector >::iterator bitend = m_bordersPerProc.end();
603 std::map<int,u32vector >::iterator bitbegin = m_bordersPerProc.begin();
604 for(std::map<int,u32vector >::iterator bit = bitbegin; bit != bitend; ++bit){
605 int key = bit->first;
606 const u32vector & pborders = bit->second;
607 size_t buffSize = 0;
608 size_t nofPbordersPerProc = pborders.size();
609 if(fixedDataSize != 0){
610 buffSize = fixedDataSize*nofPbordersPerProc;
611 }
612 else{
613 for(size_t i = 0; i < nofPbordersPerProc; ++i){
614 buffSize += userData.size(pborders[i]);
615 }
616 }
617 //enlarge buffer to store number of pborders from this proc
618 buffSize += sizeof(size_t);
619 //build buffer for this proc
620 communicator.setSend(key,buffSize);
621 SendBuffer & sendBuffer = communicator.getSendBuffer(key);
622 //store number of pborders from this proc at the begining
623 sendBuffer << nofPbordersPerProc;
624
625 //WRITE SEND BUFFERS
626 for(size_t j = 0; j < nofPbordersPerProc; ++j){
627 userData.gather(sendBuffer,pborders[j]);
628 }
629 }
630
631 communicator.discoverRecvs();
632 communicator.startAllRecvs();
633
634 communicator.startAllSends();
635
636 //READ RECEIVE BUFFERS
637 int ghostOffset = 0;
638 std::vector<int> recvRanks = communicator.getRecvRanks();
639 std::sort(recvRanks.begin(),recvRanks.end());
640 for(int rank : recvRanks){
641 communicator.waitRecv(rank);
642 RecvBuffer & recvBuffer = communicator.getRecvBuffer(rank);
643 size_t nofGhostFromThisProc = 0;
644 recvBuffer >> nofGhostFromThisProc;
645 for(size_t k = 0; k < nofGhostFromThisProc; ++k){
646 userData.scatter(recvBuffer, k+ghostOffset);
647 }
648 ghostOffset += nofGhostFromThisProc;
649 }
650 communicator.waitAllSends();
651 }

759 template<class Impl>
760 void
761 privateLoadBalance(const uint32_t *partition, DataLBInterface<Impl> *userData = nullptr){
762
763 (*m_log) << " " << std::endl;
764 if (m_serial) {
765 (*m_log) << " Initial Serial distribution : " << std::endl;
766 } else {
767 (*m_log) << " Initial Parallel partition : " << std::endl;
768 }
769 (*m_log) << " Octants for proc "+ std::to_string(static_cast<unsigned long long>(0))+" : " + std::to_string(static_cast<unsigned long long>(m_partitionRangeGlobalIdx[0]+1)) << std::endl;
770 for(int ii=1; ii<m_nproc; ii++){
771 (*m_log) << " Octants for proc "+ std::to_string(static_cast<unsigned long long>(ii))+" : " + std::to_string(static_cast<unsigned long long>(m_partitionRangeGlobalIdx[ii]-m_partitionRangeGlobalIdx[ii-1])) << std::endl;
772 }
773
774 // Compute load balance ranges
775 std::unordered_map<int, std::array<uint32_t, 2>> sendRanges = evalLoadBalanceSendRanges(partition);
776 std::unordered_map<int, std::array<uint32_t, 2>> recvRanges = evalLoadBalanceRecvRanges(partition);
777
778 m_loadBalanceRanges = LoadBalanceRanges(m_serial, sendRanges, recvRanges);
779
780 // Compute information about the new partitioning
781 assert(m_nproc > 0);
782
783 uint32_t newSizeOctants = partition[m_rank];
784
785 uint64_t newLastOctantGlobalIdx;
786 uint64_t newFirstOctantGlobalIdx;
787 if (newSizeOctants > 0) {
788 newLastOctantGlobalIdx = partition[0];
789 for (int p = 1; p < m_rank + 1; ++p) {
790 newLastOctantGlobalIdx += partition[p];
791 }
792 --newLastOctantGlobalIdx;
793
794 newFirstOctantGlobalIdx = newLastOctantGlobalIdx - newSizeOctants + 1;
795 } else {
796 newLastOctantGlobalIdx = 0;
797 newFirstOctantGlobalIdx = 1;
798 }
799
800 // Partition internal octants
801 if (m_serial) {
802 m_lastOp = OP_LOADBALANCE_FIRST;
803
804 if (newFirstOctantGlobalIdx != 0) {
805 for (uint32_t i = 0; i < newSizeOctants; ++i){
806 m_octree.m_octants[i] = m_octree.m_octants[newFirstOctantGlobalIdx + i];
807 if (userData) {
808 userData->move(newFirstOctantGlobalIdx + i, i);
809 }
810 }
811 }
812
813 m_octree.m_octants.resize(newSizeOctants);
814 m_octree.m_octants.shrink_to_fit();
815
816 if (userData) {
817 userData->resize(newSizeOctants);
818 userData->shrink();
819 }
820 } else {
821 // Compute information about the current partitioning
822 uint64_t lastOctantGlobalIdx = m_partitionRangeGlobalIdx[m_rank];
823 uint64_t firstOctantGlobalIdx = lastOctantGlobalIdx - m_octree.m_octants.size() + 1;
824
825 // Initialize communications
826 DataCommunicator lbCommunicator(m_comm);
827
828 if (!userData || userData->fixedSize()) {
829 for (const auto &entry : recvRanges) {
830 int rank = entry.first;
831 uint32_t beginRecvIdx = entry.second[0];
832 uint32_t endRecvIdx = entry.second[1];
833
834 uint32_t nOctantsToReceive = endRecvIdx - beginRecvIdx;
835 std::size_t buffSize = nOctantsToReceive * Octant::getBinarySize();
836 if (userData) {
837 buffSize += nOctantsToReceive * userData->fixedSize();
838 }
839
840 lbCommunicator.setRecv(rank, buffSize);
841 lbCommunicator.startRecv(rank);
842 }
843 }
844
845 for (const auto &entry : sendRanges) {
846 int rank = entry.first;
847 uint32_t beginSendIdx = entry.second[0];
848 uint32_t endSendIdx = entry.second[1];
849
850 uint32_t nOctantsToSend = endSendIdx - beginSendIdx;
851 std::size_t buffSize = nOctantsToSend * Octant::getBinarySize();
852 if (userData) {
853 if (userData->fixedSize()) {
854 buffSize += nOctantsToSend * userData->fixedSize();
855 } else {
856 for (uint32_t i = beginSendIdx; i < endSendIdx; ++i) {
857 buffSize += userData->size(i);
858 }
859 }
860 }
861 lbCommunicator.setSend(rank, buffSize);
862
863 SendBuffer &sendBuffer = lbCommunicator.getSendBuffer(rank);
864 for (uint32_t i = beginSendIdx; i < endSendIdx; ++i) {
865 sendBuffer << m_octree.m_octants[i];
866 userData->gather(sendBuffer, i);
867 }
868
869 lbCommunicator.startSend(rank);
870 }
871
872 if (userData && !userData->fixedSize()) {
873 lbCommunicator.discoverRecvs();
874 lbCommunicator.startAllRecvs();
875 }
876
877 // Move resident octants into place
878 if (newSizeOctants > m_octree.m_octants.size()) {
879 m_octree.m_octants.reserve(newSizeOctants);
880 m_octree.m_octants.resize(newSizeOctants);
881
882 if (userData) {
883 userData->resize(newSizeOctants);
884 }
885 }
886
887 bool hasResidentOctants;
888 if (newSizeOctants > 0) {
889 hasResidentOctants = true;
890 if (newFirstOctantGlobalIdx > lastOctantGlobalIdx) {
891 hasResidentOctants = false;
892 } else if (newLastOctantGlobalIdx < firstOctantGlobalIdx) {
893 hasResidentOctants = false;
894 }
895 } else {
896 hasResidentOctants = false;
897 }
898
899 if (hasResidentOctants) {
900 uint32_t firstResidentGlobalOctantIdx = std::max(firstOctantGlobalIdx, newFirstOctantGlobalIdx);
901 uint32_t lastResidentGlobalOctantIdx = std::min(lastOctantGlobalIdx, newLastOctantGlobalIdx);
902
903 uint32_t nofResidents = lastResidentGlobalOctantIdx - firstResidentGlobalOctantIdx + 1;
904 uint32_t newFirstResidentOffsetIdx = firstResidentGlobalOctantIdx - newFirstOctantGlobalIdx;
905 uint32_t firstResidentOffsetIdx = firstResidentGlobalOctantIdx - firstOctantGlobalIdx;
906
907 // If residents are moved closer to the head, we need to move
908 // them from the first to the last. Otherwise, if resident are
909 // moved closer to the tail, we need to move them in reversed
910 // order, i.e., from the last to the first (otherwise octants
911 // will be overwritten during the relocaiton).
912 if (newFirstResidentOffsetIdx != firstResidentOffsetIdx) {
913 for (uint32_t i = 0; i < nofResidents; ++i) {
914 int residentIdx;
915 if (newFirstResidentOffsetIdx < firstResidentOffsetIdx) {
916 residentIdx = i;
917 } else {
918 residentIdx = nofResidents - i - 1;
919 }
920
921 m_octree.m_octants[newFirstResidentOffsetIdx + residentIdx] = m_octree.m_octants[firstResidentOffsetIdx + residentIdx];
922 if (userData) {
923 userData->move(firstResidentOffsetIdx + residentIdx, newFirstResidentOffsetIdx + residentIdx);
924 }
925 }
926 }
927 }
928
929 if (newSizeOctants < m_octree.m_octants.size()) {
930 m_octree.m_octants.resize(newSizeOctants);
931 m_octree.m_octants.shrink_to_fit();
932
933 if (userData) {
934 userData->resize(newSizeOctants);
935 userData->shrink();
936 }
937 }
938
939 // Read buffers and build new octants
940 int nCompletedRecvs = 0;
941 while (nCompletedRecvs < lbCommunicator.getRecvCount()) {
942 int senderRank = lbCommunicator.waitAnyRecv();
943 RecvBuffer &recvBuffer = lbCommunicator.getRecvBuffer(senderRank);
944
945 const std::array<uint32_t, 2> &recvRange = recvRanges.at(senderRank);
946 uint32_t beginRecvIdx = recvRange[0];
947 uint32_t endRecvIdx = recvRange[1];
948 assert(userData || ((endRecvIdx - beginRecvIdx) == (recvBuffer.getSize() / (Octant::getBinarySize()))));
949 assert(!userData || !userData->fixedSize() || ((endRecvIdx - beginRecvIdx) == (recvBuffer.getSize() / (Octant::getBinarySize() + userData->fixedSize()))));
950
951 for (uint32_t i = beginRecvIdx; i < endRecvIdx; ++i) {
952 recvBuffer >> m_octree.m_octants[i];
953 if (userData) {
954 userData->scatter(recvBuffer, i);
955 }
956 }
957
958 ++nCompletedRecvs;
959 }
960
961 lbCommunicator.waitAllSends();
962 }
963
964 // Update load balance information
965 updateLoadBalance();
966
967 // Update ghosts
968 m_octree.m_ghosts.clear();
969
970 computeGhostHalo();
971
972 if (userData) {
973 userData->resizeGhost(m_octree.getNumGhosts());
974 }
975 }
976#endif
977
978 // =============================================================================== //
979
980 };
