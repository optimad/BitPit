163void ConfigParser::read(const std::string &filename, bool append)
164{
165 // Processing not-append requests
166 if (!append) {
167 clear();
168 }
169
170 // Read the configuration
171 std::string extension = "";
172 std::size_t dotPosition = filename.find_last_of(".");
173 if (dotPosition != std::string::npos) {
174 extension = filename.substr(dotPosition + 1);
175 extension = utils::string::trim(extension);
176 }
177
178 if (extension == "xml" || extension == "XML") {
179 config::XML::readConfiguration(filename, m_root, m_checkVersion, m_version, this);
180#if BITPIT_ENABLE_RAPIDJSON
181 } else if (extension == "json" || extension == "JSON") {
182 config::JSON::readConfiguration(filename, this);
183#endif
184 } else {
185 throw std::runtime_error("ConfigParser::read - Unsupported file format");
186 }
187}
188
198void ConfigParser::write(const std::string &filename) const
199{
200 // Write the configuration
201 std::string extension = "";
202 std::size_t dotPosition = filename.find_last_of(".");
203 if (dotPosition != std::string::npos) {
204 extension = filename.substr(dotPosition + 1);
205 extension = utils::string::trim(extension);
206 }
207
208 if (extension == "xml" || extension == "XML") {
209 config::XML::writeConfiguration(filename, m_root, m_version, this);
210#if BITPIT_ENABLE_RAPIDJSON
211 } else if (extension == "json" || extension == "JSON") {
212 bool prettify = true;
213 config::JSON::writeConfiguration(filename, this, prettify);
214#endif
215 } else {
216 throw std::runtime_error("ConfigParser::write - Unsupported file format");
217 }
218}
