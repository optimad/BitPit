65void readConfiguration(const std::string &filename, Config *rootConfig)
66{
67 if (!rootConfig) {
68 throw std::runtime_error("JSON::readDoc Null Config tree structure passed");
69 }
70
71 // Open the file
72 //
73 // To be compatible with Windows platforms, the file is open as a binary
74 // file (this works fine also on Posix platforms).
75 std::FILE *fp = std::fopen(filename.c_str(), "rb");
76
77 if (!fp) {
78 std::string message = "JSON:readDoc impossible to write on file : ";
79 message += filename.c_str();
80 throw std::runtime_error(message.c_str());
81 }
82
83 // Create a buffer for reading stream purposes.
84 std::vector<char> readBuffer(65536);
85 rapidjson::FileReadStream bis(fp, readBuffer.data(), readBuffer.size());
86
87 // Wrap the readstream into an auto encoding handle. This will let
88 // absorb any UTF encoded file into a regular UTF-8 document.
89 rapidjson::AutoUTFInputStream<unsigned, rapidjson::FileReadStream> eis(bis);
90
91 // Parse UTF file into UTF-8 in memory
92 rapidjson::Document jsonRoot;
93 jsonRoot.ParseStream<0, rapidjson::AutoUTF<unsigned> >(eis);
94
95 // Close the file
96 std::fclose(fp);
97
98 // Handle parse errors
99 if (jsonRoot.HasParseError()) {
100 std::string message = "JSON:readDoc error of type : ";
101 message += std::string(rapidjson::GetParseError_En(jsonRoot.GetParseError()));
102 throw std::runtime_error(message.c_str());
103 }
104
105 // Root should be an object, bitpit doens't support root arrays.
106 assert(jsonRoot.IsObject() && "JSON:readDoc parsed document root is not an object");
107
108 // Fill the configuration
109 readNode("", jsonRoot, rootConfig);
110}

137void readNode(const std::string &key, const rapidjson::Value &value, Config *config)
138{
139 if (value.IsObject()) {
140 // Get the section
141 Config::Section *section;
142 if (key.empty()) {
143 // If the key is empty we are reading the root.
144 section = config;
145 } else if (!config->isMultiSectionsEnabled() &&config->hasSection(key)) {
146 // Get the requestd section
147 section = &(config->getSection(key));
148 } else {
149 // Create a new section.
150 //
151 // If the name already exists, it will be handled as a multi-section
152 // in the tree map.
153 section = &(config->addSection(key));
154 }
155
156 // Read the contents of the section
157 for (auto itr = value.MemberBegin(); itr != value.MemberEnd(); ++itr) {
158 readNode(std::string(itr->name.GetString()), itr->value, section);
159 }
160
161 } else if (value.IsArray()) {
162 // Check if multi-section configurations are supported
163 bool multiSectionsEnabled = config->isMultiSectionsEnabled();
164 if (!multiSectionsEnabled) {
165 throw std::runtime_error("JSON:readNode reading array but Config MultiSection disabled");
166 }
167
168 // Read the array contents
169 for (auto itr = value.Begin(); itr != value.End(); ++itr) {
170 if (itr->IsObject()) {
171 readNode(key, *itr, config);
172 } else {
173 throw std::runtime_error("JSON:readNode only arrays of JSON Objects are supported!");
174 }
175 }
176
177 } else {
178 // Read a single JSON POD type (bool, numbers, strings null).
179 config->set(key, decodeValue(value));
180 }
181}

202void writeConfiguration(const std::string &filename, const Config *rootConfig, bool prettify)
203{
204 if (!rootConfig) {
205 throw std::runtime_error("JSON::writeConfiguration Null Config tree structure passed");
206 }
207
208 // DOM Document is GenericDocument<UTF8<>>
209 rapidjson::Document jsonRoot;
210
211 // Get the allocator
212 rapidjson::Document::AllocatorType &allocator = jsonRoot.GetAllocator();
213
214 // Create a root node and recursively fill it
215 jsonRoot.SetObject();
216 writeNode(rootConfig, jsonRoot, allocator);
217
218 // Write on file
219 //
220 // To be compatible with Windows platforms, the file is open as a binary
221 // file (this works fine also on Posix platforms).
222 std::FILE *fp = std::fopen(filename.c_str(), "wb");
223 if (!fp) {
224 std::string message = "JSON:writeConfiguration impossible to write on file : ";
225 message += filename.c_str();
226 throw std::runtime_error(message.c_str());
227 }
228
229 // Create a buffer for writing purposes.
230 std::vector<char> writeBuffer(65536);
231 rapidjson::FileWriteStream bos(fp, writeBuffer.data(), writeBuffer.size());
232
233 // Write the file
234 if (prettify) {
235 // Use prettyfied UTF-8 JSON format with default indentantion
236 rapidjson::PrettyWriter<rapidjson::FileWriteStream> writer(bos);
237 jsonRoot.Accept(writer);
238
239 } else {
240 // Use single-ultracompact UTF-8 JSON format
241 rapidjson::Writer<rapidjson::FileWriteStream> writer(bos);
242 jsonRoot.Accept(writer);
243 }
244
245 // Close the file.
246 std::fclose(fp);
247}

263void writeNode(const Config *config, rapidjson::Value &rootJSONData, rapidjson::Document::AllocatorType &allocator)
264{
265 // Write the options
266 for (const auto &entry : config->getOptions()) {
267 const std::string &configKey = entry.first;
268 const std::string &configValue = entry.second;
269
270 rapidjson::Value jsonKey;
271 jsonKey.SetString(configKey, allocator);
272
273 rapidjson::Value jsonValue = encodeValue(configValue, allocator);
274
275 rootJSONData.AddMember(jsonKey, jsonValue, allocator);
276 }
277
278 // Get list of unique section keys
279 std::set<std::string> listOfSectionsKeys;
280 for (const auto &entry : config->getSections()) {
281 listOfSectionsKeys.insert(entry.first);
282 }
283
284 // Write the sections
285 for (const std::string &configKey : listOfSectionsKeys) {
286 const Config::ConstMultiSection &keySections = config->getSections(configKey);
287 if (keySections.empty()) {
288 continue;
289 }
290
291 std::size_t nEntries = keySections.size();
292
293 rapidjson::Value object(rapidjson::kObjectType);
294 rapidjson::Value arrayOfObjects(rapidjson::kArrayType);
295 if (nEntries > 1) {
296 arrayOfObjects.Reserve(nEntries, allocator);
297 }
298
299 // Process section content
300 for (std::size_t i = 0; i< nEntries; ++i) {
301 writeNode(keySections[i], object, allocator);
302 if (nEntries > 1) {
303 arrayOfObjects.PushBack(object, allocator);
304 rapidjson::Value swapDummy(rapidjson::kObjectType);
305 object.Swap(swapDummy);
306 }
307 }
308
309 rapidjson::Value jsonKey;
310 jsonKey.SetString(configKey, allocator);
311
312 if (nEntries > 1) {
313 rootJSONData.AddMember(jsonKey, arrayOfObjects, allocator);
314 } else {
315 rootJSONData.AddMember(jsonKey, object, allocator);
316 }
317 }
318}

335std::string decodeValue(const rapidjson::Value &value)
336{
337 if (value.IsBool()) {
338 return std::to_string(int(value.GetBool()));
339 } else if (value.IsString()) {
340 return std::string(value.GetString());
341 } else if (value.IsNumber()) {
342 // To avoid problems, numbers are always casted to the maximum allowed
343 // datatype.
344 std::ostringstream stringStream;
345 if (value.IsUint64()) {
346 // Try uint casting first
347 stringStream << value.GetUint64();
348
349 } else if (value.IsInt64()) {
350 // Then the normal int casting
351 stringStream << value.GetInt64();
352 } else {
353 // Otherwise try it as double
354 stringStream << std::scientific << std::setprecision(8) << value.GetDouble();
355 }
356
357 return stringStream.str();
358 } else {
359 return std::string("");
360 }
361}

378rapidjson::Value encodeValue(const std::string &stringValue, rapidjson::Document::AllocatorType &allocator)
379{
380 rapidjson::Value value;
381
382 // Empty string are null values
383 if (stringValue.empty()) {
384 return value;
385 }
386
387 // Strings that contains non-numeric charactes are strings
388 std::size_t nonNumericCharacterPos = stringValue.find_first_not_of("Ee+-.0123456789");
389 if (nonNumericCharacterPos != std::string::npos) {
390 value.SetString(stringValue, allocator);
391 return value;
392 }
393
394 // Convert the string into a number
395 std::istringstream stringStream(stringValue);
396 if (stringValue.find_first_of("Ee.") != std::string::npos) {
397 // Convert the string into a double
398 double number;
399 stringStream >> number;
400 value.SetDouble(number);
401 } else if (stringValue.find_first_of("-") != std::string::npos) {
402 // Convert the string into a long
403 int64_t number;
404 stringStream >> number;
405 value.SetInt64(number);
406
407 } else {
408 // Convert the string into an unsigned long
409 uint64_t number;
410 stringStream >> number;
411 value.SetUint64(number);
412 }
413
414 return value;
415}
