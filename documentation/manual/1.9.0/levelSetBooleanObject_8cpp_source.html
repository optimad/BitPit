72void LevelSetBooleanResult::update(const LevelSetObject *object, double value)
73{
74 if( m_operation == LevelSetBooleanOperation::UNION){
75 if(m_value > value) {
76 m_object = object;
77 m_objectSign = 1;
78
79 m_value = m_objectSign * value;
80 }
81
82 } else if ( m_operation == LevelSetBooleanOperation::INTERSECTION){
83 if(m_value < value) {
84 m_object = object;
85 m_objectSign = 1;
86
87 m_value = m_objectSign * value;
88 }
89
90 } else if ( m_operation == LevelSetBooleanOperation::SUBTRACTION){
91 if(m_value < - value) {
92 m_object = object;
93 m_objectSign = -1;
94
95 m_value = m_objectSign * value;
96 }
97 }
98}

247void LevelSetBooleanObject::replaceSourceObject(const LevelSetObject *current, const LevelSetObject *updated){
248
249 std::size_t nSources = m_sourceObjects.size();
250 for (std::size_t i = 0; i < nSources; ++i) {
251 const LevelSetObject *source = m_sourceObjects[i];
252 if (source == current) {
253 m_sourceObjects[i] = updated;
254 return;
255 }
256 }
257
258 throw std::runtime_error("Unable to find the source that should be replaced.");
259}

346double LevelSetBooleanObject::getMinSurfaceFeatureSize() const {
347
348 bool minimumValid = false;
349 double minimumSize = levelSetDefaults::SIZE;
350 for( const LevelSetObject *object : m_sourceObjects ){
351 double objectMinimumSize = object->getMinSurfaceFeatureSize();
352 if (objectMinimumSize < 0) {
353 continue;
354 }
355
356 minimumValid = true;
357 minimumSize = std::min(objectMinimumSize, minimumSize);
358 }
359
360 if (!minimumValid) {
361 minimumSize = - levelSetDefaults::SIZE;
362 }
363
364 return minimumSize;
365}

412LevelSetBooleanResult LevelSetBooleanObject::computeBooleanResult( long id ) const{
413
414 // Early return if the are no objects
415 if (m_sourceObjects.empty()) {
416 return LevelSetBooleanResult( getBooleanOperation() );
417 }
418
419 // Identify informaiton about the source
420 LevelSetBooleanResult result( getBooleanOperation(), m_sourceObjects[0], m_sourceObjects[0]->getValue(id) ) ;
421 for( size_t n=1; n<m_sourceObjects.size(); ++n){
422 result.update(m_sourceObjects[n], m_sourceObjects[n]->getValue(id));
423 }
424
425 return result;
426}

433LevelSetBooleanResult LevelSetBooleanObject::computeBooleanResult( const std::array<double,3> &coords ) const{
434
435 // Early return if the are no objects
436 if (m_sourceObjects.empty()) {
437 return LevelSetBooleanResult( getBooleanOperation() );
438 }
439
440 // Identify informaiton about the source
441 LevelSetBooleanResult result( getBooleanOperation(), m_sourceObjects[0], m_sourceObjects[0]->computeLevelSetInfo(coords).value);
442 for( size_t n=1; n<m_sourceObjects.size(); ++n){
443 result.update(m_sourceObjects[n], m_sourceObjects[n]->computeLevelSetInfo( coords ).value );
444 }
445
446 return result;
447}
