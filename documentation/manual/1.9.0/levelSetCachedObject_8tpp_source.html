259template<typename narrow_band_cache_t>
260LevelSetInfo LevelSetCachedObject<narrow_band_cache_t>::getLevelSetInfo( long id)const{
261
262 const narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
263 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->find(id) ;
264 if( narrowBandCacheItr != narrowBandCache->end() ){
265 double value = narrowBandCache->getValue(narrowBandCacheItr);
266 const std::array<double, 3> &gradient = narrowBandCache->getGradient(narrowBandCacheItr);
267
268 return LevelSetInfo(value, gradient);
269 }

271 return LevelSetInfo();

310template<typename narrow_band_cache_t>
311short LevelSetCachedObject<narrow_band_cache_t>::getSign( long id ) const {
312
313 // Check if the sign can be evaluated from narrowband value
314 const narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
315 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->find(id) ;
316 if( narrowBandCacheItr != narrowBandCache->end() ){
317 double value = narrowBandCache->getValue(narrowBandCacheItr);
318
319 return evalValueSign(value);
320 }
321
322 // Check if the sign can be evaluated from the propagation
323 if (!isSignStorageDirty()) {
324 const LevelSetSignStorage *propagatedSignStorage = getSignStorage();
325 LevelSetSignStorage::KernelIterator propagatedSignStorageItr = propagatedSignStorage->find(id);
326 LevelSetSignStorage::Sign propagatedSign = propagatedSignStorage->at(propagatedSignStorageItr);
327 if (propagatedSign != LevelSetSignStorage::SIGN_UNDEFINED) {
328 return static_cast<short>(propagatedSign);
329 }
330 }
331
332 // Unable to evaluate the sign
333 //
334 // The sign cannot be evaluated, let's return the defualt sign.
335 return levelSetDefaults::SIGN;
336
337}

362template<typename narrow_band_cache_t>
363LevelSetObject::UpdateStrategy LevelSetCachedObject<narrow_band_cache_t>::getPartitioningUpdateStrategy() const {
364 // If the kernel of the cache uses an automatic synchronization, it's not possible to
365 // update the object after a partitioning exchanging data. That's because, at the time
366 // the update takes place, the automatic synchronization has already deleted the data
367 // of the cells that have been sent.
368 const narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
369 if (narrowBandCache->getKernelSyncMode() == narrow_band_cache_t::KERNEL_SYNC_MODE_MANUAL) {
370 return UPDATE_STRATEGY_EXCHANGE;
371 } else {
372 return UPDATE_STRATEGY_EVALUATE;
373 }
374}

381template<typename narrow_band_cache_t>
382void LevelSetCachedObject<narrow_band_cache_t>::pruneNarrowBand(const std::vector<long> &cellIds){
383
384 narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
385 for (long cellId : cellIds) {
386 if (narrowBandCache->contains(cellId)) {
387 narrowBandCache->erase(cellId, false);
388 }
389 }
390
391 narrowBandCache->syncStorages();
392
393}
