59template<typename narrow_band_cache_t>
60template<typename object_t>
61LevelSetImmutableObject<narrow_band_cache_t>::LevelSetImmutableObject(object_t *source)
62 : LevelSetObject(*source)
63{
64 // Mesh information
65 const VolumeKernel *mesh = (const_cast<const object_t *>(source))->getKernel()->getMesh();
66
67 VolumeKernel::CellConstIterator cellBegin = mesh->cellConstBegin();
68 VolumeKernel::CellConstIterator cellEnd = mesh->cellConstEnd();
69
70 // Fill narrow band information
71 narrow_band_cache_t *narrowBandCache = this->initializeNarrowBandCache();
72
73 bool isSignStorageNeeded = false;
74 for (VolumeKernel::CellConstIterator cellItr = cellBegin; cellItr != cellEnd; ++cellItr) {
75 long cellId = cellItr.getId();
76 bool cellInsideNarrowBand = source->isInNarrowBand(cellId);
77
78 if (cellInsideNarrowBand) {
79 double value = source->getValue(cellId);
80 std::array<double,3> gradient = source->getGradient(cellId);
81
82 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->insert(cellId);
83 narrowBandCache->set(narrowBandCacheItr, value, gradient);
84 } else if (!isSignStorageNeeded) {
85 short cellSign = source->getSign(cellId);
86 isSignStorageNeeded = (cellSign != levelSetDefaults::SIGN);
87 }
88 }
89
90 // Fill sign information
91 if (isSignStorageNeeded) {
92 LevelSetSignStorage *signStorage = initializeSignStorage();
93
94 for (VolumeKernel::CellConstIterator cellItr = cellBegin; cellItr != cellEnd; ++cellItr) {
95 long cellId = cellItr.getId();
96 std::size_t cellRawId = cellItr.getRawIndex();
97 LevelSetSignStorage::KernelIterator cellSignStorageItr = signStorage->rawFind(cellRawId);
98 signStorage->at(cellSignStorageItr) = source->getSign(cellId);
99 }
100
101 signStorage->setDirty(false);
102 }
103}

176template<typename narrow_band_cache_t>
177short LevelSetImmutableObject<narrow_band_cache_t>::getSign(long id) const {
178
179 // Check if the sign can be evaluated from narrowband value
180 const narrow_band_cache_t *narrowBandCache = this->getNarrowBandCache();
181 typename narrow_band_cache_t::KernelIterator narrowBandCacheItr = narrowBandCache->find(id);
182 if(narrowBandCacheItr != narrowBandCache->end()){
183 double value = narrowBandCache->getValue(narrowBandCacheItr);
184
185 return evalValueSign(value);
186 }
187
188 // Check if the sign can be evaluated from the storage
189 if (!isSignStorageDirty()) {
190 const LevelSetSignStorage *signStorage = getSignStorage();
191 LevelSetSignStorage::KernelIterator signStorageItr = signStorage->find(id);
192 LevelSetSignStorage::Sign storedSign = signStorage->at(signStorageItr);
193 if (storedSign != LevelSetSignStorage::SIGN_UNDEFINED) {
194 return static_cast<short>(storedSign);
195 }
196 }
197
198 // Unable to evaluate the sign
199 //
200 // The sign cannot be evaluated, let's return the defualt sign.
201 return levelSetDefaults::SIGN;
202
203}
