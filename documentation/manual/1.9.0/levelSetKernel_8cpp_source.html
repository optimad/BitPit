139void LevelSetKernel::initializeCommunicator()
140{
141 // The communicator has to be valid
142 MPI_Comm communicator = m_mesh->getCommunicator();
143 if (communicator == MPI_COMM_NULL) {
144 throw std::runtime_error ("Levelset communicator is not valid");
145 }
146
147 // Communication can be set just once
148 if (isCommunicatorSet()) {
149 int comparison;
150 MPI_Comm_compare(communicator, m_communicator, &comparison);
151 if (comparison == MPI_IDENT || comparison == MPI_CONGRUENT) {
152 return;
153 } else {
154 throw std::runtime_error ("Levelset communicator cannot be modified.");
155 }
156 }
157
158 // Create a duplicate of the patch communicator
159 MPI_Comm_dup(communicator, &m_communicator);
160}

185void LevelSetKernel::freeCommunicator() {
186
187 if (!isCommunicatorSet()) {
188 return;
189 }
190
191 int finalizedCalled;
192 MPI_Finalized(&finalizedCalled);
193 if (finalizedCalled) {
194 return;
195 }
196
197 MPI_Comm_free(&m_communicator);
198}
