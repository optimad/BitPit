162std::vector<const LevelSetObject*> LevelSetProxyObject::getPrimarySourceObjects() const{
163
164 std::vector<const LevelSetObject*> objects;
165 for( const LevelSetObject *sourceObject : getSourceObjects()){
166 if( const LevelSetProxyObject *proxySourceObject = dynamic_cast<const LevelSetProxyObject*>(sourceObject) ){
167 std::vector<const LevelSetObject*> sourcePrimarySourceObjects = proxySourceObject->getPrimarySourceObjects();
168 objects.insert(objects.end(), sourcePrimarySourceObjects.begin(), sourcePrimarySourceObjects.end());
169 } else {
170 objects.push_back(sourceObject);
171 }
172 }
173
174 return objects;
175
