370void LevelSetSegmentationKernel::setSurface( const SurfUnstructured *surface, double featureAngle){
371
372 // Check if adjacencies are built
373 if (surface->getAdjacenciesBuildStrategy() == SurfUnstructured::ADJACENCIES_NONE) {
374 throw std::runtime_error ("Segmentation needs adjacencies!") ;
375 }
376
377 // Surface information
378 m_surface = surface;
379 m_featureAngle = featureAngle;
380
381 // Segment vertices information
382 m_segmentVertexOffset.unsetKernel();
383 m_segmentVertexOffset.setStaticKernel(&m_surface->getCells());
384
385 std::size_t nTotalSegmentVertices = 0;
386 for( auto itr = m_segmentVertexOffset.begin(); itr != m_segmentVertexOffset.end(); ++itr ){
387 *itr = nTotalSegmentVertices;
388 nTotalSegmentVertices += m_surface->getCells().rawAt(itr.getRawIndex()).getVertexCount();
389 }
390
391 // Normals
392 m_segmentNormalsValid.unsetKernel();
393 m_segmentNormalsValid.setStaticKernel(&m_surface->getCells());
394 m_segmentNormalsValid.fill(false);
395 m_segmentNormalsStorage.unsetKernel();
396 m_segmentNormalsStorage.setStaticKernel(&m_surface->getCells());
397
398 m_unlimitedVertexNormalsValid.unsetKernel();
399 m_unlimitedVertexNormalsValid.setStaticKernel(&m_surface->getVertices());
400 m_unlimitedVertexNormalsValid.fill(false);
401 m_unlimitedVertexNormalsStorage.unsetKernel();
402 m_unlimitedVertexNormalsStorage.setStaticKernel(&m_surface->getVertices());
403
404 m_limitedSegmentVertexNormalValid.assign(nTotalSegmentVertices, false);
405
406 // Initialize search tree
407 m_searchTree = std::unique_ptr<SurfaceSkdTree>(new SurfaceSkdTree(surface));
408 m_searchTree->build();
409}

429int LevelSetSegmentationKernel::getSegmentInfo( const std::array<double,3> &pointCoords, long segmentId, bool signd, double &distance, std::array<double,3> &gradient, std::array<double,3> &normal ) const {
430
431 // Segment information
432 SurfUnstructured::CellConstIterator segmentIterator = m_surface->getCellConstIterator(segmentId);
433 const Cell &segment = *segmentIterator ;
434 ElementType segmentType = segment.getType();
435 ConstProxyVector<long> segmentVertexIds = segment.getVertexIds() ;
436 int nSegmentVertices = segmentVertexIds.size() ;
437
438 // Projct the point on the surface and evaluate the point-projeciont vector
439 BITPIT_CREATE_WORKSPACE(lambda, double, nSegmentVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
440 std::array<double,3> pointProjectionVector = pointCoords;
441 switch (segmentType) {
442
443 case ElementType::VERTEX :
444 {
445 long id = segmentVertexIds[0] ;
446 pointProjectionVector -= m_surface->getVertexCoords(id);
447
448 break;
449 }
450
451 case ElementType::LINE:
452 {
453 long id0 = segmentVertexIds[0] ;
454 long id1 = segmentVertexIds[1] ;
455 pointProjectionVector -= CGElem::projectPointSegment( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), lambda);
456
457 break;
458 }
459
460 case ElementType::TRIANGLE:
461 {
462 long id0 = segmentVertexIds[0] ;
463 long id1 = segmentVertexIds[1] ;
464 long id2 = segmentVertexIds[2] ;
465 pointProjectionVector -= CGElem::projectPointTriangle( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), m_surface->getVertexCoords(id2), lambda );
466
467 break;
468 }
469
470 default:
471 {
472 ConstProxyVector<long> elementVertexIds = m_surface->getFacetOrderedVertexIds(segment);
473 BITPIT_CREATE_WORKSPACE(segmentVertexCoors, std::array<double BITPIT_COMMA 3>, nSegmentVertices, ReferenceElementInfo::MAX_ELEM_VERTICES);
474 m_surface->getVertexCoords(elementVertexIds.size(), elementVertexIds.data(), segmentVertexCoors);
475 pointProjectionVector -= CGElem::projectPointPolygon( pointCoords, nSegmentVertices, segmentVertexCoors, lambda );
476
477 break;
478 }
479
480 }
481
482 // Compute surface normal
483 normal = computeSurfaceNormal(segmentIterator, lambda);
484
485 // Evaluate distance from surface
486 distance = norm2(pointProjectionVector);
487
488 // Check if the point lies on the segmentation
489 //
490 // If the distance is zero, the point and the projection are coincident,
491 // this means that the point lies on the segmentation.
492 double distanceTolerance = m_surface->getTol();
493 bool pointOnSegmentation = utils::DoubleFloatingEqual()(distance, 0., distanceTolerance, distanceTolerance);
494
495 // Evaluate levelset gradient
496 if (!pointOnSegmentation) {
497 gradient = pointProjectionVector / distance;
498 } else {
499 if (signd) {
500 gradient = normal;
501 } else {
502 gradient = {{0., 0., 0.}};
503 }
504 }
505
506 // Evaluate levelset sign
507 //
508 // The sign is computed by determining the side of the point with respect
509 // to the normal plane. The sign will be zero if the point lies exaclty
510 // on the segmentation or on the normal plane. In the latter case the sign
511 // must be evaluated taking into account the the curvature of the surface.
512 // However, this is not yet implemented.
513 //
514 // The sign should be evaluated with the same tolerance used when checking
515 // if the point lies on the segmentation.
516 std::array<double, 3> pseudoNormal = computePseudoNormal(segmentIterator, lambda);
517 double pointProjectionNormalComponent = dotProduct(pointProjectionVector, pseudoNormal);
518
519 int s;
520 if (utils::DoubleFloatingEqual()(pointProjectionNormalComponent, 0., distanceTolerance, distanceTolerance)) {
521 s = 0;
522 } else if (pointProjectionNormalComponent > 0) {
523 s = 1;
524 } else {
525 s = -1;
526 }
527
528 if (!pointOnSegmentation && s == 0) {
529 distance = levelSetDefaults::VALUE;
530 gradient = levelSetDefaults::GRADIENT;
531 normal = levelSetDefaults::GRADIENT;
532
533 return 1;
534 }
535
536 // Use sign to update levelset information
537 //
538 // If signed distance are computed, the distance value and gradient
539 // need to be changed accordingly. If unsigned distance are computed
540 // the orientation of the suraface normal is discarded and in order
541 // to agnostic with repect the two sides of the surface
542 if (s < 0) {
543 distance *= (double) ( signd *s + (!signd) *1);
544 gradient *= (double) ( signd *s + (!signd) *1);
545 normal *= (double) ( signd *1 + (!signd) *s);
546 }
547
548 return 0;
549}

631std::array<double,3> LevelSetSegmentationKernel::computePseudoNormal( const SurfUnstructured::CellConstIterator &segmentIterator, const double *lambda ) const {
632
633 // Early return if the segment is a point
634 const Cell &segment = *segmentIterator;
635 ElementType segmentType = segment.getType();
636 if (segmentType == ElementType::VERTEX) {
637 return {{0., 0., 0.}};
638 }
639
640 // Evaluate pseudo normal
641 int positionFlag;
642 if (segmentType == ElementType::LINE) {
643 positionFlag = CGElem::convertBarycentricToFlagSegment(lambda, m_surface->getTol());
644 } else {
645 int nSegmentVertices = segment.getVertexCount();
646 positionFlag = CGElem::convertBarycentricToFlagPolygon(nSegmentVertices, lambda, m_surface->getTol());
647 if (positionFlag > 0) {
648 int polygonVertex = positionFlag - 1;
649 int elementVertex = m_surface->getFacetOrderedLocalVertex(segment, polygonVertex);
650
651 positionFlag = elementVertex + 1;
652 } else if (positionFlag < 0) {
653 int polygonEdge = (- positionFlag) - 1;
654 int elementEdge = m_surface->getFacetOrderedLocalEdge(segment, polygonEdge);
655
656 positionFlag = - (elementEdge + 1);
657 }
658 }
659
660 std::array<double,3> pseudoNormal;
661 if (positionFlag == 0) {
662 pseudoNormal = computeSegmentNormal(segmentIterator);
663 } else if (positionFlag > 0) {
664 int vertex = positionFlag - 1;
665 pseudoNormal = computeSegmentVertexNormal(segmentIterator, vertex, false);
666 } else if (positionFlag < 0) {
667 int edge = (- positionFlag) - 1;
668 pseudoNormal = computeSegmentEdgeNormal(segmentIterator, edge);
669 }
670
671 return pseudoNormal;
672}

686std::array<double,3> LevelSetSegmentationKernel::computeSurfaceNormal( const SurfUnstructured::CellConstIterator &segmentIterator, const double *lambda ) const {
687
688 // Early return if the segment is a point
689 const Cell &segment = *segmentIterator;
690 ElementType segmentType = segment.getType();
691 if (segmentType == ElementType::VERTEX) {
692 return {{0., 0., 0.}};
693 }
694
695 // Evaluate surface normal
696 std::size_t nSegmentVertices = segment.getVertexCount();
697 std::array<double,3> surfaceNormal = lambda[0] * computeSegmentVertexNormal(segmentIterator, 0, true);
698 for (std::size_t i = 1; i < nSegmentVertices; ++i) {
699 surfaceNormal += lambda[i] * computeSegmentVertexNormal(segmentIterator, i, true);
700 }
701 surfaceNormal /= norm2(surfaceNormal);
702
703 return surfaceNormal;
704}

735std::array<double,3> LevelSetSegmentationKernel::computeSegmentEdgeNormal( const SurfUnstructured::CellConstIterator &segmentIterator, int edge ) const {
736
737 std::array<double,3> normal = computeSegmentNormal(segmentIterator);
738
739 if (segmentIterator->getAdjacencyCount(edge) > 0) {
740 long neighId = segmentIterator->getAdjacency(edge);

745 }
746

761std::array<double,3> LevelSetSegmentationKernel::computeSegmentVertexNormal( const SurfUnstructured::CellConstIterator &segmentIterator, int vertex, bool limited ) const {
762
763 // Segment information
764 long segmentId = segmentIterator.getId();
765 long segmentRawId = segmentIterator.getRawIndex();
766 const Cell &segment = *segmentIterator;
767
768 // Update the cache
769 //
770 // Cache is updated for all the vertices of the segment.
771 long vertexId = segment.getVertexId(vertex);
772 std::size_t vertexRawId = m_surface->getVertices().getRawIndex(vertexId);
773 bool hasUnlimitedNormal = m_unlimitedVertexNormalsValid.rawAt(vertexRawId);
774
775 bool hasLimitedNormal = m_limitedSegmentVertexNormalValid[m_segmentVertexOffset.rawAt(segmentRawId) + vertex];
776
777 if (!hasUnlimitedNormal || !hasLimitedNormal) {
778 static std::vector<long> vertexNeighbours;
779 vertexNeighbours.clear();
780 m_surface->findCellVertexNeighs(segmentId, vertex, &vertexNeighbours);
781
782 std::array<double, 3> limitedVertexNormal;
783 std::array<double, 3> unlimitedVertexNormal;
784 if (hasUnlimitedNormal) {
785 limitedVertexNormal = m_surface->evalLimitedVertexNormal(segmentId, vertex, vertexNeighbours.size(), vertexNeighbours.data(), m_featureAngle) ;
786 unlimitedVertexNormal = m_unlimitedVertexNormalsStorage.rawAt(vertexRawId);
787 } else {
788 m_surface->evalVertexNormals(segmentId, vertex, vertexNeighbours.size(), vertexNeighbours.data(), m_featureAngle, &unlimitedVertexNormal, &limitedVertexNormal) ;
789 }
790
791 // Store vertex limited normal
792 //
793 // Both limited and unlimited normal are evaluated, however limited
794 // normal is only stored if its misalignment with respect to the
795 // unlimited normal is greater than a defined tolerance.
796 if( !hasLimitedNormal ){
797 double misalignment = norm2(unlimitedVertexNormal - limitedVertexNormal) ;
798 if( misalignment >= m_surface->getTol() ){
799 std::pair<long, int> segmentVertexKey = std::make_pair(segmentId, vertex);
800 m_limitedSegmentVertexNormalStorage.insert({segmentVertexKey, std::move(limitedVertexNormal)}) ;
801 }
802 m_limitedSegmentVertexNormalValid[m_segmentVertexOffset.rawAt(segmentRawId) + vertex] = true;
803 }
804
805 // Store vertex unlimited normal
806 if ( !hasUnlimitedNormal ) {
807 m_unlimitedVertexNormalsStorage.rawAt(vertexRawId) = std::move(unlimitedVertexNormal);
808 m_unlimitedVertexNormalsValid.rawAt(vertexRawId) = true ;
809 }
810 }
811
812 // Get the normal from the cache
813 if (limited) {
814 std::pair<long, int> segmentVertexKey = std::make_pair(segmentId, vertex);
815 auto limitedSegmentVertexNormal = m_limitedSegmentVertexNormalStorage.find(segmentVertexKey);
816 if (limitedSegmentVertexNormal != m_limitedSegmentVertexNormalStorage.end()) {
817 return limitedSegmentVertexNormal->second;
818 }
819 }
820
821 return m_unlimitedVertexNormalsStorage.rawAt(vertexRawId);
822}
