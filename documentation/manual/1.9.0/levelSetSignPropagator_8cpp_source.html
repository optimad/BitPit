101void LevelSetSignPropagator::execute(const std::vector<adaption::Info> &adaptionData, LevelSetSignedObjectInterface *object)
102{
103 // Get the storage
104 //
105 // If the storage is not initialized yet, the sign should be evaluated
106 // from scratch.
107 LevelSetSignStorage *storage = object->getSignStorage();
108 if (!storage) {
109 execute(object);
110 return;
111 }
112
113 // Initialize sign propagation
114 //
115 // We need to check if propagation is really needed and initialize stored sign of new cells.
116 // Propagation is needed if the stored sign is marked as dirty or if cells has been modified
117 // by mesh adaption.
118 bool propagationNeeded = storage->isDirty();
119 for (const adaption::Info &adaptionInfo : adaptionData) {
120 if (adaptionInfo.entity != adaption::Entity::ENTITY_CELL) {
121 continue;
122 }
123
124 propagationNeeded = true;
125
126 for (long cellId : adaptionInfo.current) {
127 LevelSetSignStorage::KernelIterator cellStorageItr = storage->find(cellId);
128 storage->at(cellStorageItr) = LevelSetSignStorage::SIGN_UNDEFINED;
129 }
130 }
131
132#if BITPIT_ENABLE_MPI
133 if (m_mesh->isPartitioned()) {
134 MPI_Allreduce(MPI_IN_PLACE, &propagationNeeded, 1, MPI_C_BOOL, MPI_LOR, m_mesh->getCommunicator());
135 }
136#endif
137
138 // Early return if propagation is not needed
139 if (!propagationNeeded) {
140 return;
141 }
142
143#if BITPIT_ENABLE_MPI
144 // Initialize sign on ghost cells
145 //
146 // Ghost cells are not tracked by adaption, we need to explicitly initialize the sign
147 // on all ghost cells.
148 VolumeKernel::CellConstIterator cellGhostBegin = m_mesh->ghostCellConstBegin();
149 VolumeKernel::CellConstIterator cellGhostEnd = m_mesh->ghostCellConstEnd();
150
151 for (VolumeKernel::CellConstIterator cellItr = cellGhostBegin; cellItr != cellGhostEnd; ++cellItr) {
152 std::size_t cellRawId = cellItr.getRawIndex();
153 LevelSetSignStorage::KernelIterator cellStorageItr = storage->rawFind(cellRawId);
154 storage->at(cellStorageItr) = LevelSetSignStorage::SIGN_UNDEFINED;
155 }
156#endif
157
158 // Propagate sign
159 propagate(object, storage);
160}

171void LevelSetSignPropagator::propagate(const LevelSetObjectInterface *object, LevelSetSignStorage *storage)
172{
173 // Sign propagation does not work for Cartesian meshes in memory-light mode.
174 if (VolCartesian *cartesianMesh = dynamic_cast<VolCartesian *>(m_mesh)) {
175 if (cartesianMesh->getMemoryMode() == VolCartesian::MEMORY_LIGHT) {
176 throw std::runtime_error("Sign propagation does not work for Cartesian meshes in memory-light mode.");
177 }
178 }
179
180 // Initialize propagation information
181 initializePropagation(object);
182
183 // Set sign of cells in the narrowband
184 //
185 // Cells in the narrowband will defined the seed for the propagation.
186 VolumeKernel::CellConstIterator cellBegin = m_mesh->cellConstBegin();
187 VolumeKernel::CellConstIterator cellEnd = m_mesh->cellConstEnd();
188
189 std::vector<std::size_t> rawSeeds;
190 for (VolumeKernel::CellConstIterator cellItr = cellBegin; cellItr != cellEnd; ++cellItr) {
191 std::size_t cellRawId = cellItr.getRawIndex();
192 LevelSetSignStorage::KernelIterator cellSignStorageItr = storage->rawFind(cellRawId);
193 int cellSign = storage->at(cellSignStorageItr);
194 if (cellSign == LevelSetSignStorage::SIGN_UNDEFINED) {
195 long cellId = cellItr.getId();
196 if (object->isInNarrowBand(cellId)) {
197 cellSign = object->getSign(cellId);
198 }
199 }
200
201 if (cellSign != LevelSetSignStorage::SIGN_UNDEFINED) {
202 setSign(cellRawId, cellSign, storage);
203 rawSeeds.push_back(cellRawId);
204 }
205 }
206
207 // Check if there are seeds to be used for sign propagation
208 //
209 // We need at least one seed to be able to perform sign propagation (i.e., at least one
210 // cell should be in the levelset narrowband).
211 long nGlobalSeeds = rawSeeds.size();
212 long nGlobalWaiting = m_nWaiting;
213#if BITPIT_ENABLE_MPI
214 if (m_mesh->isPartitioned()) {
215 MPI_Allreduce(MPI_IN_PLACE, &nGlobalSeeds, 1, MPI_LONG, MPI_SUM, m_mesh->getCommunicator());
216 MPI_Allreduce(MPI_IN_PLACE, &nGlobalWaiting, 1, MPI_LONG, MPI_SUM, m_mesh->getCommunicator());
217 }
218#endif
219
220 if (nGlobalWaiting != 0 && nGlobalSeeds == 0) {
221 throw std::runtime_error("Unable to propagate the sign: the list of seeds is empty!");
222 }
223
224 // Use the seeds to propagate the sign
225 executeSeedPropagation(rawSeeds, storage);
226
227#if BITPIT_ENABLE_MPI
228 // If there are cells with an unknown sign, data communication among
229 // ghost cells is needed. However it is only possibly to have cells with
230 // an unknown sign for partinioned patches.
231 nGlobalWaiting = m_nWaiting;
232 if (m_mesh->isPartitioned()) {
233 MPI_Allreduce(MPI_IN_PLACE, &nGlobalWaiting, 1, MPI_LONG, MPI_SUM, m_mesh->getCommunicator());
234 }
235
236 if (nGlobalWaiting != 0) {
237 assert(m_mesh->isPartitioned());
238 assert(m_mesh->getProcessorCount() != 1);
239
240 // Initialize the communicator for exchanging the sign of the ghosts
241 DataCommunicator dataCommunicator(m_mesh->getCommunicator());
242
243 signed char exchangedSign;
244 std::size_t exchangedDataSize = sizeof(exchangedSign);
245
246 // Set the receives
247 for (const auto &entry : m_mesh->getGhostCellExchangeTargets()) {
248 const int rank = entry.first;
249 const auto &list = entry.second;
250
251 dataCommunicator.setRecv(rank, list.size() * exchangedDataSize);
252 }
253
254 // Set the sends
255 for (const auto &entry : m_mesh->getGhostCellExchangeSources()) {
256 const int rank = entry.first;
257 auto &list = entry.second;
258
259 dataCommunicator.setSend(rank, list.size() * exchangedDataSize);
260 }
261
262 // Communicate sign information among the partitions
263 while (nGlobalWaiting != 0) {
264 // Start the receives
265 for (const auto &entry : m_mesh->getGhostCellExchangeTargets()) {
266 const int rank = entry.first;
267 dataCommunicator.startRecv(rank);
268 }
269
270 // Start the sends
271 for (const auto &entry : m_mesh->getGhostCellExchangeSources()) {
272 const int rank = entry.first;
273 const auto &sendIds = entry.second;
274 SendBuffer &buffer = dataCommunicator.getSendBuffer(rank);
275
276 for (long cellId : sendIds) {
277 exchangedSign = LevelSetSignStorage::SIGN_UNDEFINED;
278 if (m_propagationStates.at(cellId) == STATE_REACHED) {
279 LevelSetSignStorage::KernelIterator exchangedSignStorageItr = storage->find(cellId);
280 exchangedSign = storage->at(exchangedSignStorageItr);
281 }
282 buffer << exchangedSign;
283 }
284
285 dataCommunicator.startSend(rank);
286 }
287
288 // Receive the sign and propagate the sign
289 //
290 // If we discover the sign of a ghost, we can use it as a seed.
291 rawSeeds.clear();
292 int nCompletedRecvs = 0;
293 while (nCompletedRecvs < dataCommunicator.getRecvCount()) {
294 int rank = dataCommunicator.waitAnyRecv();
295 const auto &recvIds = m_mesh->getGhostCellExchangeTargets(rank);
296 RecvBuffer &buffer = dataCommunicator.getRecvBuffer(rank);
297
298 // Receive data and detect new seeds
299 for (long cellId : recvIds) {
300 buffer >> exchangedSign;
301 if (exchangedSign == LevelSetSignStorage::SIGN_UNDEFINED) {
302 continue;
303 }
304
305 VolumeKernel::CellConstIterator cellItr = m_mesh->getCells().find(cellId);
306 std::size_t cellRawId = cellItr.getRawIndex();
307 PropagationState cellPropagationState = m_propagationStates.rawAt(cellRawId);
308 if (cellPropagationState == STATE_WAITING) {
309 setSign(cellRawId, exchangedSign, storage);
310 rawSeeds.push_back(cellRawId);
311 } else if (cellPropagationState == STATE_REACHED) {
312 assert(storage->at(storage->find(cellId)) == exchangedSign);
313 }
314 }
315
316 ++nCompletedRecvs;
317 }
318
319 if (rawSeeds.size() > 0) {
320 executeSeedPropagation(rawSeeds, storage);
321 }
322
323 // Wait to the sends to finish
324 dataCommunicator.waitAllSends();
325
326 // Update the global counter for cells with an unknow sign
327 nGlobalWaiting = m_nWaiting;
328 MPI_Allreduce(MPI_IN_PLACE, &nGlobalWaiting, 1, MPI_LONG, MPI_SUM, m_mesh->getCommunicator());
329 }
330 }
331
332 // Communicate the sign of the external region
333 //
334 // The sign has to be consistent among all the partitions.
335 bool exchangeExternalSign;
336 if (m_mesh->isPartitioned()) {
337 exchangeExternalSign = (m_externalSign != LevelSetSignStorage::SIGN_UNDEFINED);
338 MPI_Allreduce(MPI_IN_PLACE, &exchangeExternalSign, 1, MPI_C_BOOL, MPI_LOR, m_mesh->getCommunicator());
339 } else {
340 exchangeExternalSign = false;
341 }
342
343 if (exchangeExternalSign) {
344 bool positiveExternalSign = (m_externalSign == 1);
345 MPI_Allreduce(MPI_IN_PLACE, &positiveExternalSign, 1, MPI_C_BOOL, MPI_LOR, m_mesh->getCommunicator());
346
347 bool negativeExternalSign = (m_externalSign == -1);
348 MPI_Allreduce(MPI_IN_PLACE, &negativeExternalSign, 1, MPI_C_BOOL, MPI_LOR, m_mesh->getCommunicator());
349
350 if (positiveExternalSign && negativeExternalSign) {
351 m_externalSign = LevelSetSignStorage::SIGN_UNDEFINED;
352 } else if (positiveExternalSign) {
353 m_externalSign = 1;
354 } else if (negativeExternalSign) {
355 m_externalSign = -1;
356 } else {
357 m_externalSign = LevelSetSignStorage::SIGN_UNDEFINED;
358 }
359 }
360#else
361 // Check that the sign has been propagated into all regions
362 assert(m_nWaiting == 0);
363#endif
364
365 // Assign the sign to the external cells
366 if (m_nExternal > 0) {
367 // Check if the sign of the external region has been identified
368 if (m_externalSign == LevelSetSignStorage::SIGN_UNDEFINED) {
369 throw std::runtime_error("Sign of external region not properly identified!");
370 }
371
372 // Assign the sign to the cells of the external region
373 for (auto cellItr = cellBegin; cellItr != cellEnd; ++cellItr) {
374 std::size_t cellRawId = cellItr.getRawIndex();
375 if (m_propagationStates.rawAt(cellRawId) != STATE_EXTERNAL) {
376 continue;
377 }
378
379 setSign(cellRawId, m_externalSign, storage);
380 }
381 }
382
383 // Finalize propagation
384 finalizePropagation(storage);
385}

392void LevelSetSignPropagator::initializePropagation(const LevelSetObjectInterface *object)
393{
394 // Initialize propagation state
395 m_nWaiting = m_mesh->getCellCount();
396
397 m_propagationStates.unsetKernel();
398 m_propagationStates.setStaticKernel(&(m_mesh->getCells()));
399 m_propagationStates.fill(STATE_WAITING);
400
401 // Identify external cells
402 //
403 // A cell is external if it is completely outside the object bounding box.
404 // If the cell may be intersected by the object (i.e., the bounding box of
405 // the cell intersects the bounding box of the object), the cell cannot be
406 // flagged as external.
407 const LevelSetBoundedObject *boundedObject = dynamic_cast<const LevelSetBoundedObject *>(object);
408
409 if (boundedObject) {
410 // Evaluate the bounding box of the object
411 //
412 // The current process may only have the portion of the object needed for
413 // evaluating the levelset on the cells of its mesh, therefore we need to
414 // evaluate the overall bounding box across all process.
415 std::array<double,3> objectBoxMin;
416 std::array<double,3> objectBoxMax;
417#if BITPIT_ENABLE_MPI
418 boundedObject->getGlobalBoundingBox(objectBoxMin, objectBoxMax);
419#else
420 boundedObject->getBoundingBox(objectBoxMin, objectBoxMax);
421#endif
422
423 // Check if the bounding box of the object is empty
424 bool isObjectBoxEmpty = false;
425 for (int i = 0; i < 3; ++i) {
426 if (objectBoxMax[i] < objectBoxMin[i]) {
427 isObjectBoxEmpty = true;
428 break;
429 }
430 }
431
432 // Identify cell outside the bounding box of the object
433 if (!isObjectBoxEmpty) {
434 // Get tolerance for distance comparison
435 double distanceTolerance = m_mesh->getTol();
436
437 // Check if the patch intersects the bounding box of the object
438 std::array<double,3> patchBoxMin;
439 std::array<double,3> patchBoxMax;
440 m_mesh->getBoundingBox(patchBoxMin, patchBoxMax);
441
442 bool isPatchIntersected = CGElem::intersectBoxBox(patchBoxMin, patchBoxMax, objectBoxMin, objectBoxMax, 3, distanceTolerance);
443
444 // Detect external cells
445 VolumeKernel::CellConstIterator cellBegin = m_mesh->cellConstBegin();
446 VolumeKernel::CellConstIterator cellEnd = m_mesh->cellConstEnd();
447
448 m_nExternal = 0;
449 for (VolumeKernel::CellConstIterator itr = cellBegin; itr != cellEnd; ++itr) {
450 // Cells inside the narrowband cannot be external
451 long cellId = itr.getId();
452 if (object->isInNarrowBand(cellId)) {
453 continue;
454 }
455
456 // Check if the centroid is inside the bounding box
457 //
458 // Cells with the centroid inside the bounding box of the object
459 // cannot be external cells
460 if (isPatchIntersected) {
461 double geometricTolerance = m_mesh->getTol();
462 std::array<double, 3> cellCentroid = object->getKernel()->computeCellCentroid(cellId);
463
464 bool isCentroidInternal = true;
465 for (int i = 0; i < 3; ++i) {
466 if (cellCentroid[i] < objectBoxMin[i] - geometricTolerance || cellCentroid[i] > objectBoxMin[i] + geometricTolerance) {
467 isCentroidInternal = false;
468 break;
469 }
470 }
471
472 if (isCentroidInternal) {
473 continue;
474 }
475 }
476
477 // Check if the cell is inside the bounding box of the object
478 std::array<double,3> cellBoxMin;
479 std::array<double,3> cellBoxMax;
480 m_mesh->evalCellBoundingBox(cellId, &cellBoxMin, &cellBoxMax);
481
482 bool isCellIntersected = CGElem::intersectBoxBox(cellBoxMin, cellBoxMax, objectBoxMin, objectBoxMax, 3, distanceTolerance);
483 if (isCellIntersected) {
484 continue;
485 }
486
487 std::size_t cellRawId = itr.getRawIndex();
488 m_propagationStates.rawAt(cellRawId) = STATE_EXTERNAL;
489 ++m_nExternal;
490 }
491
492 // Initialize the sign of the external region
493 m_externalSign = LevelSetSignStorage::SIGN_UNDEFINED;
494 } else {
495 // All cells are external
496 m_nExternal = m_mesh->getCellCount();
497
498 // Initialize the sign of the external region
499 m_externalSign = LevelSetSignStorage::SIGN_POSITIVE;
500 }
501
502 // Update the number of cells waiting for sing propagation
503 m_nWaiting -= m_nExternal;
504 } else {
505 // It's not possible to identify
506 m_nExternal = 0;
507
508 // Initialize the sign of the external region
509 m_externalSign = LevelSetSignStorage::SIGN_UNDEFINED;
510 }

528void LevelSetSignPropagator::executeSeedPropagation(const std::vector<std::size_t> &rawSeeds, LevelSetSignStorage *storage)
529{
530 const PiercedVector<Cell, long> &meshCells = m_mesh->getCells();
531
532 std::vector<std::size_t> rawProcessList;
533
534 std::size_t nRawSeeds = rawSeeds.size();
535 while (nRawSeeds != 0) {
536 // Get a seed
537 --nRawSeeds;
538 std::size_t seedRawId = rawSeeds[nRawSeeds];
539
540 // Get the sign of the seed
541 LevelSetSignStorage::KernelIterator seedSignStorageItr = storage->rawFind(seedRawId);
542 LevelSetSignStorage::Sign seedSign = storage->at(seedSignStorageItr);
543 assert(seedSign >= -1 && seedSign <= 1);
544
545 // Initialize the process list with the seed
546 rawProcessList.resize(1);
547 rawProcessList[0] = seedRawId;
548
549 // Propagate the sign
550 while (!rawProcessList.empty()) {
551 std::size_t cellRawId = rawProcessList.back();
552 rawProcessList.resize(rawProcessList.size() - 1);
553
554 // Set the sign of the cell
555 //
556 // We need to set the sign only if it has not already been set
557 // (for example, the sign of the seeds is already set).
558 LevelSetSignStorage::KernelIterator cellSignStorageItr = storage->rawFind(cellRawId);
559 LevelSetSignStorage::Sign cellSign = storage->at(cellSignStorageItr);
560 if (cellSign == LevelSetSignStorage::SIGN_UNDEFINED) {
561 setSign(cellRawId, seedSign, storage);
562 }
563
564 // Process cell neighbours
565 //
566 // If a neighbour is waiting for the propagation, add it to the
567 // process list. When the propagation reaches an external cell
568 // the sign of the seed frow which the propagation started will
569 // be the sign of the external region.
570 const Cell &cell = meshCells.rawAt(cellRawId);
571 const long *cellNeighs = cell.getAdjacencies();
572 int nCellNeighs = cell.getAdjacencyCount();
573 for(int n = 0; n < nCellNeighs; ++n){
574 long neighId = cellNeighs[n];
575 VolumeKernel::CellConstIterator neighItr = meshCells.find(neighId);
576 std::size_t neighRawId = neighItr.getRawIndex();
577
578 PropagationState neighState = m_propagationStates.rawAt(neighRawId);
579 if (neighState == STATE_WAITING) {
580 rawProcessList.push_back(neighRawId);
581 } else if (neighState == STATE_EXTERNAL) {
582 // If the sign of the external region is unknown it can
583 // be assigned, otherwise check if the current sign is
584 // consistent with the previously evaluated sign.
585 if (m_externalSign == LevelSetSignStorage::SIGN_UNDEFINED) {
586 m_externalSign = seedSign;
587 } else if (m_externalSign != seedSign) {
588 throw std::runtime_error("Mismatch in sign of external region!");
589 }
590 }
591 }
592
593 // Check if the propagation is complete
594 //
595 // It can be possible to stop the propagation without processing
596 // all the cells in the process list if:
597 // - all cells have been reached by the propagation;
598 // - the sign of the external region have been identified.
599 bool emptyWaitingList = (m_nWaiting == 0);
600 bool externalSignIdentified = (m_nExternal == 0) || (m_externalSign != LevelSetSignStorage::SIGN_UNDEFINED);
601 if (emptyWaitingList && externalSignIdentified) {
602 break;
603 }
604 }
605 }

616 m_propagationStates.unsetKernel(false);
617
618 // The propagated sign is now up-to-date
619 storage->setDirty(false);

629void LevelSetSignPropagator::setSign(std::size_t cellRawId, LevelSetSignStorage::Sign cellSign, LevelSetSignStorage *storage)
630{
631 assert(cellSign >= -1 && cellSign <= 1);
632
633 // Set the sign of the cell
634 LevelSetSignStorage::KernelIterator cellStorageItr = storage->rawFind(cellRawId);
635 storage->at(cellStorageItr) = cellSign;
636
637 // Update the state of the cell
638 //
639 // If the cell is external, the sign of the eternal region should be
640 // updated.
641 //
642 // If the cell is waiting for the propagation, the counter of the
643 // waiting cell list should be updated.
644 PropagationState *cellState = m_propagationStates.rawData(cellRawId);
645 if (*cellState == STATE_EXTERNAL) {
646 if (m_externalSign == LevelSetSignStorage::SIGN_UNDEFINED) {
647 m_externalSign = cellSign;
648 } else if (m_externalSign != cellSign) {
649 throw std::runtime_error("Mismatch in sign of external region!");
650 }
651 } else if (*cellState == STATE_WAITING) {
652 --m_nWaiting;
653 }
654 *cellState = STATE_REACHED;
655}
