280LevelSetInternalPiercedStorageManager::KernelIterator LevelSetInternalPiercedStorageManager::insert(long id, bool sync)
281{
282 Kernel::FillAction action = m_internalKernel.fillHead(id);
283 if (sync) {
284 syncStorages();
285 }
286
287 return m_internalKernel.rawFind(action.info[PiercedSyncAction::INFO_POS]);
288}
