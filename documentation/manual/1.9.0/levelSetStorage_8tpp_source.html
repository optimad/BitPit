792template<typename kernel_t, typename kernel_iterator_t>
793void LevelSetStorageManager<kernel_t, kernel_iterator_t>::write(const std::vector<long> &ids, SendBuffer &buffer)
794{
795 // Evaluate the size of the buffer
796 std::size_t nBufferItems = 0;
797 for (long id : ids) {
798 if (contains(id)) {
799 ++nBufferItems;
800 }
801 }
802
803 std::size_t bufferSize = buffer.getSize() + sizeof(std::size_t);
804 for (const auto &storage : m_storages) {
805 bufferSize += nBufferItems * storage.second->getItemBinarySize();
806 }
807 buffer.setSize(bufferSize);
808
809 // Fill the buffer
810 buffer << nBufferItems;
811 for (std::size_t k = 0; k < ids.size(); ++k) {
812 // Get an iterator pointing to the item
813 long id = ids[k];
814 KernelIterator itr = find(id);
815 if (itr == end()) {
816 continue;
817 }
818
819 // Write the index of the cell
820 buffer << k;
821
822 // Write item data
823 for (const auto &storage : m_storages) {
824 storage.second->writeItem(itr, buffer);
825 }
826 }
827}

838template<typename kernel_t, typename kernel_iterator_t>
839void LevelSetStorageManager<kernel_t, kernel_iterator_t>::read(const std::vector<long> &ids, RecvBuffer &buffer)
840{
841 std::size_t nReceviedItems;
842 buffer >> nReceviedItems;
843
844 for (std::size_t i = 0; i < nReceviedItems; ++i) {
845 // Read the id of the item
846 std::size_t k;
847 buffer >> k;
848 long id = ids[k];
849
850 // Create the item
851 KernelIterator itr = find(id);
852 if (itr == end()) {
853 itr = insert(id);
854 }
855
856 // Read item data
857 for (auto &storage : m_storages) {
858 storage.second->readItem(itr, buffer);
859 }
860 }
861}
