114void LevelSet::setMesh( VolumeKernel* mesh ) {
115
116 // Mesh can be set only once
117 if (m_kernel) {
118 throw std::runtime_error ("Mesh can be set just once.");
119 }
120
121 // Create the kernel
122 if (VolCartesian *cartesian = dynamic_cast<VolCartesian *>(mesh)) {
123 m_kernel = std::unique_ptr<LevelSetKernel>(new LevelSetCartesianKernel(*cartesian));
124 } else if (VolOctree *octree = dynamic_cast<VolOctree *>(mesh)) {
125 m_kernel = std::unique_ptr<LevelSetKernel>(new LevelSetOctreeKernel(*octree));
126 } else if (VolUnstructured *unstructured = dynamic_cast<VolUnstructured*>(mesh)) {
127 m_kernel = std::unique_ptr<LevelSetKernel>(new LevelSetUnstructuredKernel(*unstructured));
128 } else {
129 throw std::runtime_error ("Unable to create the levelset kernel. Mesh type non supported.");
130 }
131
132# if BITPIT_ENABLE_MPI
133 // Initialize the communicator
134 if (m_kernel->getMesh()->isPartitioned()) {
135 m_kernel->initializeCommunicator();
136 }
137# endif
138
139 for( auto &obj : m_objects){
140 obj.second->setKernel(m_kernel.get());
141 }
142
143}

321int LevelSet::registerObject( std::unique_ptr<LevelSetObject> &&object ) {
322
323 int id = object->getId();
324 if (id == levelSetDefaults::OBJECT) {
325 id = m_objectIdentifierGenerator.generate();
326 object->setId(id);
327 } else {
328 m_objectIdentifierGenerator.setAssigned(id);
329 }
330
331 if( m_kernel){
332 object->setKernel(m_kernel.get());
333 }
334
335 if (m_narrowBandSize > 0.) {
336 object->setSizeNarrowBand(m_narrowBandSize);
337 }
338
339 m_objects[id] = std::move(object) ;
340
341 setObjectProcessingOrder(id);
342
343 incrementObjectsReferenceCount(id);
344
345 return id;
346}

421void LevelSet::makeObjectImmutable( int id ) {
422
423 LevelSetObject *object = getObjectPtr( id ) ;
424
425 // Get the list of referring objects
426 //
427 // Referring objects are objects for which this object is a source.
428 std::vector<int> referringProxyObjectIds;
429 if( object->getReferenceCount() > 0 ) {
430 for( const auto &entry : m_objects ) {
431 const LevelSetProxyObject *proxyObject = dynamic_cast<const LevelSetProxyObject*>(entry.second.get());
432 if (!proxyObject) {
433 continue;
434 }
435
436 std::vector<const LevelSetObject *> sourceObjects = proxyObject->getSourceObjects() ;
437 std::size_t nSourceObjects = sourceObjects.size() ;
438 for ( std::size_t i = 0; i < nSourceObjects; ++i ){
439 const LevelSetObject *sourceObject = sourceObjects[i];
440 if (sourceObject == object) {
441 referringProxyObjectIds.push_back(proxyObject->getId());
442 }
443 }
444 }
445 }
446
447 // Create the immutable object
448 std::unique_ptr<LevelSetObject> immutableObject = LevelSetObjectFactory::createImmutableObject<LevelSetImmutableNarrowBandCache>( m_kernel.get(), m_storageType, object ) ;
449 assert( immutableObject );
450
451 // Update referring objects
452 for (int referringProxyObjectId : referringProxyObjectIds) {
453 LevelSetProxyObject *referringProxyObject = static_cast<LevelSetProxyObject *>(getObjectPtr( referringProxyObjectId ));
454 referringProxyObject->replaceSourceObject(object, immutableObject.get());
455 }
456
457 // Replace the object with the newly created immutable object
458 removeObject(id, true);
459 addObject(std::move(immutableObject));
460}

474void LevelSet::setObjectProcessingOrder( int id ) {
475
476 // Immutable objects are not processed.
477 if (dynamic_cast<const LevelSetImmutableObjectBase*>(getObjectPtr(id))) {
478 return;
479 }
480
481 // Define the processing order for the object
482 //
483 // The insertion order determines the processing order, however priority
484 // is given to primary objects.
485 std::vector<int>::iterator processingOrderItr;
486 if(getObjectPtr(id)->isPrimary()){
487 std::vector<int>::iterator processingOrderBegin = m_objectsProcessingOrder.begin();
488 std::vector<int>::iterator processingOrderEnd = m_objectsProcessingOrder.end();
489
490 for (processingOrderItr = processingOrderBegin; processingOrderItr != processingOrderEnd; ++processingOrderItr) {
491 int candidateId = *processingOrderItr ;
492 const LevelSetObject *candidateObject = getObjectPtr(candidateId) ;
493 if( !candidateObject->isPrimary() ){
494 break;
495 }
496 }
497 } else {
498 processingOrderItr = m_objectsProcessingOrder.end();
499 }
500
501 m_objectsProcessingOrder.insert(processingOrderItr,id) ;
502
503}

511void LevelSet::unsetObjectProcessingOrder(int id){
512
513 // Immutable objects are not processed.
514 if (dynamic_cast<const LevelSetImmutableObjectBase*>(getObjectPtr(id))) {
515 return;
516 }
517
518 // Remove the object from the list of processed objects
519 std::vector<int>::iterator processingOrderBegin = m_objectsProcessingOrder.begin();
520 std::vector<int>::iterator processingOrderEnd = m_objectsProcessingOrder.end();
521
522 std::vector<int>::iterator processingOrderItr = std::find(processingOrderBegin, processingOrderEnd, id);
523 assert(processingOrderItr != processingOrderEnd);
524 m_objectsProcessingOrder.erase(processingOrderItr);
525
526}

724void LevelSet::compute( const std::unordered_set<LevelSetObject *> &objectProcessList ){
725
726 assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::compute()");
727
728 std::unique_ptr<LevelSetSignPropagator> signPropagator ;
729 if (m_propagateSign) {
730 signPropagator = m_kernel->createSignPropagator() ;
731 }
732
733 for( int id : m_objectsProcessingOrder){
734 LevelSetObject *object = m_objects.at(id).get() ;
735 if (objectProcessList.count(object) == 0) {
736 continue;
737 }
738
739 LevelSetSignedObjectInterface *signPropagationObject;
740 if (m_propagateSign) {
741 signPropagationObject = dynamic_cast<LevelSetSignedObjectInterface *>(object);
742 } else {
743 signPropagationObject = nullptr;
744 }
745
746 // Clear the object
747 object->clear();
748
749 // Compute levelset inside the narrowband
750 object->computeNarrowBand(m_signedDistance) ;
751#if BITPIT_ENABLE_MPI
752 object->exchangeGhosts();
753#endif
754
755 // Propagate sign
756 if(signPropagationObject) {
757 signPropagator->execute(signPropagationObject);
758 }
759 }
760}

824void LevelSet::update( const std::vector<adaption::Info> &adaptionData, const std::unordered_set<LevelSetObject *> &objectProcessList ){
825
826 assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::partition()");
827
828#if BITPIT_ENABLE_MPI
829 const VolumeKernel *mesh = m_kernel->getMesh();
830 bool isMeshPartitioned = mesh->isPartitioned();
831#endif
832
833#if BITPIT_ENABLE_MPI
834 // Set the communicator
835 //
836 // The mesh may have been partitioned after being set.
837 if (isMeshPartitioned && !m_kernel->isCommunicatorSet()) {
838 m_kernel->initializeCommunicator();
839 }
840#endif
841
842 // Check if the levelset needs to be updated
843 bool isDirty = false;
844 for( const adaption::Info &adaptionInfo : adaptionData){
845 if( adaptionInfo.entity != adaption::Entity::ENTITY_CELL){
846 continue;
847 }
848
849 isDirty = true;
850 break;
851 }
852
853#if BITPIT_ENABLE_MPI
854 if (isMeshPartitioned) {
855 MPI_Allreduce(MPI_IN_PLACE, &isDirty, 1, MPI_C_BOOL, MPI_LOR, m_kernel->getCommunicator());
856 }
857#endif
858
859 // Early return if no update is needed
860 if (!isDirty) {
861 return;
862 }
863
864 // Update kernel
865 m_kernel->update( adaptionData ) ;
866
867 // Create sign propagator
868 std::unique_ptr<LevelSetSignPropagator> signPropagator ;
869 if (m_propagateSign) {
870 signPropagator = m_kernel->createSignPropagator() ;
871 }
872
873 // Update objects
874 for( int id : m_objectsProcessingOrder){
875 LevelSetObject *object = m_objects.at(id).get() ;
876 if (objectProcessList.count(object) == 0) {
877 continue;
878 }
879
880 LevelSetSignedObjectInterface *signPropagationObject;
881 if (m_propagateSign) {
882 signPropagationObject = dynamic_cast<LevelSetSignedObjectInterface *>(object);
883 } else {
884 signPropagationObject = nullptr;
885 }
886
887 // Propagated sign is now dirty
888 //
889 // The propagated sign will be set as non-dirty by the sign propagator.
890 if (signPropagationObject) {
891 signPropagationObject->setSignStorageDirty(true);
892 }
893
894 // Update object
895 object->update( adaptionData, m_signedDistance );
896
897 // Propagate sign
898 //
899 // It's not possible to communicate sign information, therefore sign
900 // needs to be propagated also when the mesh is only partitioned.
901 if (signPropagationObject) {
902 signPropagator->execute(adaptionData, signPropagationObject);
903 }
904 }
905
906}

946std::unordered_set<LevelSetObject *> LevelSet::getObjectProcessList(std::size_t nObjects, const int *objectIds) const{
947
948 std::unordered_set<LevelSetObject *> objectProcessList;
949 for (std::size_t i = 0; i < nObjects; ++i) {
950 int objectId = objectIds[i] ;
951 LevelSetObject *object = getObjectPtr(objectId) ;
952 if (objectProcessList.count(object) != 0) {
953 continue;
954 }
955
956 objectProcessList.insert(object);
957 if( const LevelSetProxyObject *proxyObject = dynamic_cast<const LevelSetProxyObject*>(object) ){
958 std::vector<int> sourceObjectIds = proxyObject->getSourceObjectIds();
959 std::unordered_set<LevelSetObject *> sourceObjectProcessList = getObjectProcessList(sourceObjectIds.size(), sourceObjectIds.data()) ;
960 objectProcessList.insert(sourceObjectProcessList.begin(), sourceObjectProcessList.end());
961 }
962 }
963
964 return objectProcessList;
