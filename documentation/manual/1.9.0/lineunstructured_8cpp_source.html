261unsigned short LineUnstructured::importDGF(const std::string &dgf_name, int PIDOffset, bool PIDSquash)
262{
263 // ====================================================================== //
264 // VARIABLES DECLARATION //
265 // ====================================================================== //
266
267 // Local variables
268 DGFObj dgf_in(dgf_name);
269 int nV = 0, nS = 0;
270 long vcount, idx;
271 std::vector<std::array<double, 3>> vertex_list;
272 std::vector<std::vector<int>> simplex_list;
273 std::vector<int> simplex_PID;
274 std::vector<long> vertex_map;
275 std::vector<long> connect;
276
277 // Counters
278 std::vector<std::array<double, 3>>::const_iterator v_, ve_;
279 std::vector<std::vector<int>>::iterator c_, ce_;
280 std::vector<int>::iterator i_, ie_;
281 std::vector<long>::iterator j_, je_;
282
283 // ====================================================================== //
284 // IMPORT DATA //
285 // ====================================================================== //
286
287 // Read vertices and cells from DGF file
288 dgf_in.load(nV, nS, vertex_list, simplex_list, simplex_PID);
289
290 // Add vertices
291 ve_ = vertex_list.cend();
292 vcount = 0;
293 vertex_map.resize(nV);
294 for (v_ = vertex_list.cbegin(); v_ != ve_; ++v_) {
295 idx = addVertex(*v_)->getId();
296 vertex_map[vcount] = idx;
297 ++vcount;
298 } //next v_
299
300 // Update connectivity infos
301 ce_ = simplex_list.end();
302 for (c_ = simplex_list.begin(); c_ != ce_; ++c_) {
303 ie_ = c_->end();
304 for (i_ = c_->begin(); i_ != ie_; ++i_) {
305 *i_ = vertex_map[*i_];
306 } //next i_
307 } //next c_
308
309 // Add cells
310 int k;
311 for (c_ = simplex_list.begin(), k = 0; c_ != ce_; ++c_, ++k) {
312 // Create cell
313 i_ = c_->begin();
314 connect.resize(c_->size(), Vertex::NULL_ID);
315 je_ = connect.end();
316 for (j_ = connect.begin(); j_ != je_; ++j_) {
317 *j_ = *i_;
318 ++i_;
319 } //next j_
320 CellIterator cellIterator = addCell(getDGFFacetType(c_->size()), connect);
321
322 // Set cell PID
323 int cellPID = PIDOffset;
324 if (!PIDSquash) {
325 cellPID += simplex_PID[k];
326 }
327
328 cellIterator->setPID(cellPID);
329 } //next c_
330
331 return 0;
332}

341unsigned short LineUnstructured::exportDGF(const std::string &dgf_name)
342{
343 // ====================================================================== //
344 // VARIABLES DECLARATION //
345 // ====================================================================== //
346
347 // Local variables
348 DGFObj dgf_in(dgf_name);
349 int nV = getVertexCount(), nS = getCellCount();
350 int v, nv;
351 long vcount, ccount, idx;
352 std::vector<std::array<double, 3>> vertex_list(nV);
353 std::vector<std::vector<int>> simplex_list(nS);
354 std::unordered_map<long, long> vertex_map;
355
356 // Counters
357 VertexIterator v_, ve_;
358 CellIterator c_, ce_;
359
360 // ====================================================================== //
361 // EXPORT DATA //
362 // ====================================================================== //
363
364 // Create vertex list
365 ve_ = vertexEnd();
366 vcount = 0;
367 for (v_ = vertexBegin(); v_ != ve_; ++v_) {
368 idx = v_->getId();
369 vertex_list[vcount] = v_->getCoords();
370 vertex_map[idx] = vcount;
371 ++vcount;
372 } //next v_
373
374 // Add cells
375 ce_ = cellEnd();
376 ccount = 0;
377 for (c_ = cellBegin(); c_ != ce_; ++c_) {
378 ConstProxyVector<long> cellVertexIds = c_->getVertexIds();
379 nv = cellVertexIds.size();
380 simplex_list[ccount].resize(nv);
381 for (v = 0; v < nv; ++v) {
382 simplex_list[ccount][v] = vertex_map[cellVertexIds[v]];
383 } //next v
384 ++ccount;
385 } //next c_
386
387 // Read vertices and cells from DGF file
388 dgf_in.save(nV, nS, vertex_list, simplex_list);
389
390 return 0;
391}
