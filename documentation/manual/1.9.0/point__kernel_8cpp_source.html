186double PointKernel::evalPointsDistance(long id1, long id2) const
187{
188 const Cell &cell1 = m_cells[id1];
189 const Cell &cell2 = m_cells[id2];
190
191 ConstProxyVector<long> cellVertexIds1 = cell1.getVertexIds();
192 const Vertex &vertex1 = getVertex(cellVertexIds1[0]);
193
194 ConstProxyVector<long> cellVertexIds2 = cell2.getVertexIds();
195 const Vertex &vertex2 = getVertex(cellVertexIds2[0]);
196
197 double distance = norm2(vertex2.getCoords() - vertex1.getCoords());
198
199 return distance;

209std::array<double, 3> PointKernel::evalPointsDirection(long id1, long id2) const
210{
211 const Cell &cell1 = m_cells[id1];
212 const Cell &cell2 = m_cells[id2];
213
214 ConstProxyVector<long> cellVertexIds1 = cell1.getVertexIds();
215 const Vertex &vertex1 = getVertex(cellVertexIds1[0]);
216
217 ConstProxyVector<long> cellVertexIds2 = cell2.getVertexIds();
218 const Vertex &vertex2 = getVertex(cellVertexIds2[0]);
219
220 std::array<double, 3> direction = vertex2.getCoords() - vertex1.getCoords();
221
222 double distance = norm2(direction);
223 if (!utils::DoubleFloatingEqual()(distance, 0., getTol())) {
224 direction /= distance;
225 }
226
227 return direction;
228}
