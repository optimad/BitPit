567bool RBFKernel::removeData(std::vector<int> & list)
568{
569 std::set<int> setList;
570 for(auto && id : list) setList.insert(id);
571
572 int extracted = 0;
573 for(auto && id : setList) {
574 if(id>=0 && id <m_fields){
575 m_fields--;
576 int index = id-extracted;
577 assert(index >= 0);
578 if (m_mode == RBFMode::INTERP) {
579 assert((std::size_t) index < m_value.size());
580 auto valueItr = m_value.begin() + index;
581 m_value.erase(valueItr);
582 } else {
583 assert((std::size_t) index < m_weight.size());
584 auto weightItr = m_weight.begin() + index;
585 m_weight.erase(weightItr);
586 }
587
588 extracted++;
589 }
590 }
591
592 return(extracted == (int)(list.size()));
593}

614std::vector<double> RBFKernel::evalRBF( const std::array<double,3> &point)
615{
616 std::vector<double> values(m_fields, 0.);
617 int i, j;
618 double dist, basis;
619
620 for( i=0; i<m_nodes; ++i ){
621 if( m_activeNodes[i] ) {
622 dist = calcDist(point, i) / getSupportRadius(i);
623 basis = evalBasis( dist );
624
625 for( j=0; j<m_fields; ++j) {
626 values[j] += basis * m_weight[j][i];
627 }
628 }
629 }
630
631 // If INTERP mode add the polynomial contribution
632 if (m_mode == RBFMode::INTERP && m_polyEnabled) {
633 for (j = 0; j < m_fields; ++j) {
634 double *polynomialCoefficients = m_polynomial.getCoefficients(j);
635 std::vector<double> basis = evalPolynomialBasis(point);
636 int z = 0;
637 for (int iactive : m_polyActiveBasis) {
638 values[j] += polynomialCoefficients[iactive] * basis[z];
639 z++;
640 }
641 }
642 }
643
644 return values;
645}

655std::vector<double> RBFKernel::evalRBF(int jnode)
656{
657 std::vector<double> values(m_fields, 0.);
658 int i, j;
659 double dist, basis;
660
661 if(jnode<0 || jnode>= m_nodes ) {
662 return values;
663 }
664
665 for( i=0; i<m_nodes; ++i ) {
666 if( m_activeNodes[i] ) {
667 dist = calcDist(jnode, i) / getSupportRadius(i);
668 basis = evalBasis( dist );
669
670 for( j=0; j<m_fields; ++j) {
671 values[j] += basis * m_weight[j][i];
672 }
673 }
674 }
675
676 // If INTERP mode add the polynomial contribution
677 if (m_mode == RBFMode::INTERP && m_polyEnabled) {
678 for (j = 0; j < m_fields; ++j) {
679 double *polynomialCoefficients = m_polynomial.getCoefficients(j);
680 std::vector<double> basis = evalPolynomialBasis(jnode);
681 int z = 0;
682 for (int iactive : m_polyActiveBasis) {
683 values[j] += polynomialCoefficients[iactive] * basis[z];
684 z++;
685 }
686 }
687 }
688
689 return values;
690}

699int RBFKernel::solve()
700{
701 if(m_mode == RBFMode::PARAM) {
702 return -1;
703 }
704
705 if (m_polyEnabled) {
706 // Initialize polynomial
707 initializePolynomial();
708
709 // Check which parameter of the polynomial has to be activated
710 // in order to avoid undetermined system
711 initializePolynomialActiveBasis();
712 }
713
714 double dist;
715
716 int nActive = getActiveCount();
717 int nPoly = m_polyEnabled ? m_polyActiveBasis.size() : 0;
718 int nS = nActive + nPoly;
719 int nrhs = getDataCount();
720
721 int lda = nS;
722 int ldb = nS;
723 int info;
724
725 std::vector<int> ipiv(nS);
726
727 std::vector<int> activeSet( getActiveSet() );
728
729 std::vector<double> A(lda * nS);
730 std::vector<double> b(ldb * nrhs);
731
732 for (int j = 0; j < nrhs; ++j) {
733 for (const auto &i : activeSet) {
734 int k = j * ldb + i;
735 b[k] = m_value[j][i];
736 }
737 }
738
739 int k = 0;
740 for (const auto &i : activeSet) {
741 for (const auto &j : activeSet) {
742 dist = calcDist(j, i) / getSupportRadius(i); //order by column!
743 int row = k % nActive;
744 int col = k / nActive;
745 A[(col * nS) + row] = evalBasis(dist);
746 k++;
747 }
748 }
749
750 // Filling terms given by the polynomial block.
751 // Symmetric terms set in the loop.
752 if (m_polyEnabled) {
753 for (int i = 0; i < nActive; ++i) {
754 std::vector<double> polynomialTerms = evalPolynomialBasis(activeSet[i]);
755 int j = 0;
756 for (const double &val : polynomialTerms) {
757 A[(j + nActive) * nS + i] = val;
758 A[i * nS + (j + nActive)] = val;
759 j++;
760 }
761 }
762 }
763
764 info = LAPACKE_dgesv(LAPACK_COL_MAJOR, nS, nrhs, A.data(), lda, ipiv.data(), b.data(), ldb);
765
766 if( info > 0 ) {
767 printf( "The diagonal element of the triangular factor of the linear system matrix \n" );
768 printf( "U(%i,%i) is zero, so that matrix is singular;\n", info, info );
769 printf( "the solution could not be computed.\n" );
770 return 1;
771 }
772
773 m_weight.resize(nrhs);
774
775 for (int j = 0; j < nrhs; ++j) {
776 m_weight[j].resize(m_nodes, 0);
777 int k = 0;
778 for (const auto &i : activeSet) {
779 m_weight[j][i] = b[j * ldb + k];
780 ++k;
781 }
782 }
783
784 if (m_polyEnabled) {
785 for (int j = 0; j < nrhs; ++j) {
786 double *polynomialCoefficients = m_polynomial.getCoefficients(j);
787 int i = 0;
788 for (int iactive : m_polyActiveBasis) {
789 polynomialCoefficients[iactive] = b[j * ldb + nActive + i];
790 i++;
791 }
792 }
793 }
794 return 0;
795}

804int RBFKernel::greedy( double tolerance)
805{
806 if(m_mode == RBFMode::PARAM) return -1;
807
808 int i, j;
809 double error(1.e18);
810 std::vector<double> local(m_fields);
811
812 m_error.resize(m_nodes);
813
814 for( auto && active : m_activeNodes )
815 active = false;
816
817 for( i=0; i<m_nodes; ++i){
818
819 for( j=0; j<m_fields; ++j) {
820 local[j] = m_value[j][i];
821 }
822
823 m_error[i] = norm2(local);
824 }
825
826 std::ios::fmtflags streamFlags(log::cout().flags());
827
828 int errorFlag = 0;
829 while( error > tolerance) {
830 i = addGreedyPoint();
831
832 if( i != -1) {
833 m_activeNodes[i] = true;
834
835 //solve();
836 solveLSQ();
837
838 error = evalError();
839
840 log::cout() << std::scientific;
841 log::cout() << " error now " << error << " active nodes" << getActiveCount() << " / " << m_nodes << std::endl;
842 } else {
843 errorFlag = 1;
844 break;
845 }
846 }
847
848 log::cout().flags(streamFlags);
849
850 return errorFlag;
851}

926{
927 if(m_mode == RBFMode::PARAM) {
928 return -1;
929 }
930
931 int i(0), index(-1);
932 double maxError(0.);
933
934 std::vector<int> active( getActiveSet() );
935
936 for( auto error : m_error ){
937
938 if(!m_activeNodes[i] ){
939
940 if( error > maxError ) {
941 maxError = error;
942 index = i;
943 }
944 }
945
946 ++i;
947 }
948

957double RBFKernel::evalError( )
958{
959 if(m_mode == RBFMode::PARAM) {
960 return -1.0;
961 }
962
963 int i(0), j(0);
964 //int index;
965 double maxError(0), relError, realValue;
966 std::vector<double> reconValues;
967
968 for( int iX=0; iX< m_nodes; ++iX ) {
969 reconValues = evalRBF( iX );
970
971 j=0;
972 relError = 0.;
973 for( auto &val : reconValues ) {
974 realValue = m_value[j][i];
975 relError += std::pow( (val - realValue), 2 );
976
977 ++j;
978 }
979
980 relError = sqrt(relError);
981 m_error[i] = relError;
982
983 if( relError > maxError ) {
984 maxError = relError;
985 }
986
987 ++i;
988 }
989
990 return maxError;
991}

999int RBFKernel::solveLSQ()
1000{
1001 if(m_mode == RBFMode::PARAM) {
1002 return -1;
1003 }
1004
1005 if (m_polyEnabled) {
1006 // Initialize polynomial
1007 initializePolynomial();
1008
1009 // Check which parameter of the polynomial has to be activated
1010 // in order to avoid undetermined system
1011 initializePolynomialActiveBasis();
1012 }
1013
1014 double dist;
1015
1016 int nActive = getActiveCount();
1017 int nPoly = m_polyEnabled ? getPolynomialWeightsCount() : 0;
1018 int nS = nActive + nPoly;
1019 int nP = m_nodes;
1020 int nrhs = getDataCount();
1021
1022 std::vector<int> activeSet( getActiveSet() );
1023
1024 int n ,m, lda, ldb, info, rank;
1025 double rcond = -1.0;
1026
1027 m = nP;
1028 n = nS;
1029
1030 lda = m + nPoly;
1031 ldb = m + nPoly;
1032
1033 std::vector<double> A(lda * n);
1034 std::vector<double> b(ldb * nrhs, 0.0);
1035 std::vector<double> s(m);
1036
1037 for (int j = 0; j < nrhs; ++j) {
1038 for (int i = 0; i < m; ++i) {
1039 int k = j * ldb + i;
1040 b[k] = m_value[j][i];
1041 }
1042 }
1043
1044 int k = 0;
1045 for (const auto &i : activeSet) {
1046 for (int j = 0; j < nP; ++j) {
1047 dist = calcDist(j, i) / getSupportRadius(i); //order by column!
1048 int row = k % nP;
1049 int col = k / nP;
1050 A[(col * ldb) + row] = evalBasis(dist);
1051 k++;
1052 }
1053 }
1054
1055 // Filling column terms given by the polynomial block.
1056 if (m_polyEnabled) {
1057 // Loop on columns
1058 for (int i = 0; i < nActive; ++i) {
1059 std::vector<double> polynomialTerms = evalPolynomialBasis(activeSet[i]);
1060 int j = 0;
1061 for (const double &val : polynomialTerms) {
1062 A[lda * i + (nP + j)] = val;
1063 j++;
1064 }
1065 }
1066 }
1067
1068 // Filling row terms given by the polynomial block.
1069 if (m_polyEnabled) {
1070 // Loop on rows
1071 for (int i = 0; i < nP; ++i) {
1072 std::vector<double> polynomialTerms = evalPolynomialBasis(i);
1073 int j = 0;
1074 for (const double &val : polynomialTerms) {
1075 A[(lda * (nActive + j) + i)] = val;
1076 j++;
1077 }
1078 }
1079 }
1080
1081 info = LAPACKE_dgelsd(LAPACK_COL_MAJOR, lda, nS, nrhs, A.data(), lda, b.data(), ldb, s.data(), rcond, &rank);
1082
1083 if (info > 0) {
1084 return 1;
1085 }
1086
1087 m_weight.resize(nrhs);
1088
1089 for (int j = 0; j < nrhs; ++j) {
1090 m_weight[j].clear();
1091 m_weight[j].resize(m_nodes,0);
1092
1093 k=0;
1094 for( const auto &i : activeSet ) {
1095 m_weight[j][i] = b[j*ldb+k];
1096 ++k;
1097 }
1098 }
1099
1100 if (m_polyEnabled) {
1101 for (int j = 0; j < nrhs; ++j) {
1102 double *polynomialCoefficients = m_polynomial.getCoefficients(j);
1103 int i = 0;
1104 for (int iactive : m_polyActiveBasis) {
1105 polynomialCoefficients[iactive] = b[j * ldb + nActive + i];
1106 i++;
1107 }
1108 }
1109 }
1110 return 0;
1111 }
